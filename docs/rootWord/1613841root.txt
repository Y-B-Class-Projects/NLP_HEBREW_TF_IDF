ר יכריז : תוקפא הבנייה ב יודיע ערב כי הבנייה הישראלית גדה תוקפא משך חודש . יופעלו סנקציה נגד ראשי רשות מעורב חריגה בנייה , יורחקו מפרי סדר יוגשו יותר כתב אישום נגד מבצעי הבנייה בלתי ח ממשלה נימין נתניהו יכריז היום רביעי על הקפאה הבנייה התנחלות משך עשרה חודש . הקפאה לא תכלול בנייה צורך גידול טבעי בנייה שטח ירושלים . נתניהו ימסור הודעה עניין תום ישיבה של קבינט . מהלך מתואם עם שר הביטחון אהוד ברק , ככל הנראה גם תואם עם נשיא ארץ הברית ברק אובמה . לשכה ראש ממשלה נמסר בתגובה פרסום כי " חלק מאמץ הנעה שיחה שלום עם הרשות פלסטיני קידום אינטרס לאומי מקיפים של ישראל , ראש ממשלה יבקש היום קבינט אשר השהיה זמני של אישור בנייה חדש מגורים התחלות בנייה חדש מגורים יהודה ושומרון משך חודש " לפי ההערכות , הכרזה זו של נתניהו עשה סייע חיזוק מעמד של י הרשות פלסטיני אבו מאזן , ייחוד על רקע משא ומ מתקדם עם חמאס על עסק שחרור חייל חטוף גלעד שליט . אם העסקה תצא פועל , ייזקף שחרור אות אסיר פלסטיני זכות של חמאס . הקפאה בנייה לא תכלול בן ציבור מוסד חינוך בנייה מזרח ירושלים . על פי סמך מתאם פעולה ממשלה שטח מפרט את הדרכים אמצע שמר על הקפאה הבנייה התנחלות , משרד הביטחון יפקח על כל בנייה חריג התנחלות אמצעות גיחה צילום , מעקב פענוח של יהיה אחריות . כמו כן , מתאם פעולה שטח טוען כי יידרשו משאבים רב קיום וים חדש והוא צר על כך תקן חדש עדיין לא מולא . מסגרת מאמץ אכיפה על הבנייה התנחלות , מתאם פתח נוהל סילוק של " פלישה טרייה מכל סוג קרקע ב , כולל אדמה בעלות פלסטיני פרטי " הוא מוסיף שיש " לס את נושא תיחום מאחז בלתי חוקי , הפעיל סנקציה מול ראשי יישוב מועצה אשר עורב אופן ישיר ביצוע חריגה תכנוניות חריג בנייה " כמו כן , מתאם על מעצר הרחקה צעיר אשר עורב חריגה הבנייה ו מפרי סדר סדרתיים מטרידים " נוסף , מסמך עולה כי יש לייעל את תהליך אכיפה פלילי נגד מבצעי בנייה בלתי חוקי להגיש יותר כתב אישום נגדם . מפעלי יביל קרוואן יופללו יסגרו חלק אכיפה הבנייה בלתי חוקי שטח . אולם , עניין עתירה הוגש נגד מאחז בלתי חוקי הוקם בעיקר על אדמה פלסטיני פרטיות , טוען מתאם כי " יש הודיע בית המשפט שב הזה , בשל אילוץ מדיני ו מול אמריקני , אנו בקשים דחייה הריסה מאחז בלתי חוקי " ר יכריז : תוקפא הבנייה ב יודיע ערב כי הבנייה
