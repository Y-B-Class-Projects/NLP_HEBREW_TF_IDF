ה הוא רק : נתניהו פתח את דלת קלינטון ו חמשה ימי ביקור של ראש ממשלה בן יורק רחפה אל זהות של אדם ישב , או תשב , בית לבן החל ינואר . פגישה עם ני מועמד נשיאות ניסה נתניהו שמר על איזון , אך בעוד קלינטון שידר ממלכתיות , טראמפ מנסה לגרוף ר ראשון מזדקר אל מול עין של יוצאים בניין ה שדרה הראשונה בן יורק הוא גורד שחק על קומה , עלה , כמו על רב מגדל רחבי עיר , מתנוסס שם של דונלד טראמפ . יחסי קומה של מגדל מזכירות כללי של ה , דובר גוליית , מאפיל מעט על בניין מרשים . מי מחפש משל מגדל , גם ביקור שנתי של ראש ממשלה עצרת כללי של ה יש נמשל דומה צלה של התקופה סיום כהונה אובמה השאלה מי יחליף או תחליף אותו בית לבן ריחף על חמשה ימי שהות של ראש ממשלה בן יורק . סוגיה טראמפ פעם איל נ יום מועמד רפובליקני ססגוני שנוי מחלוקת אפיל אפילו על נאום חוצב להבה של נתניהו ב . זה יוצא , זה או זאת ייכנסו , היה עצם פתיח סיומת של ביקור : ביקור של נתניהו נפתח מפגש האחרון עם הנשיא אובמה , הסתיים פגישה עם ני מועמד טראמפ הילרי קלינטון , פינה זמן פגוש את נתניהו ים קמפיין טרוף מיוחד יממה לפני עימות ראשון ב . עוד אותו ע נתניהו : אתנגד החלטה מועצה הביטחון של ה טראמפ : " הילרי קלינטון אליטיסטית ו ראשון פגישה עם נתניהו : שבוע גורלי של טראמפ קלינטון טראמפ הוא זה הכתיב את לוח זמן : אחרי פמליית נתניהו אותת הוא ישמח היפגש עם מי מועמד מידה יביעו כך עניין , מטה טראמפ הר את הכפפה . בשם איזון , אנשי נתניהו מיד פנה למטה קלינטון בבקשה לתאם פגישה דומה . נתניהו אינו מנהיג מזרח תיכוני יחיד נפגש עם ני מועמד , והוא גם לא ראשון : מידה רב , ני מפגש של נשיא צר , עבד , עם קלינטון טראמפ שול עצרת ה שבוע שעבר הניח את קרקע פגישה של נתניהו אתמול . אבל יחס ונים של נתניהו עם פוליטיקה אמריקני בשנים האחרונות עמדה של ישראל תוך מציבים אותו מעמד קצת שונה . לשכה ראש ממשלה עושים מאמץ חודש אחרון להתרחק קמפיין הבחירות נשיאות לשמור על מראית עין של איזון . זאת אחרי טראומה של חירות , אז הואשם התערבות בוטה טובה מועמד רפובליקני יט רומני . עד כה נראה נתניהו משתדל עד מאוד ליישם את פתגם " מי נכווה רותחין נזהר צוננין " התקשורת הורחק שתי פגישה , תועד על יד צלם סטילס רשמי בלבד . הודעה הוציא לשכה ראש ממשלה על שתי פגישה היה זהות ניסוח , כלל ידע אינפורמטיבי בלבד : מיקום פגישה , שך הזמן היא נערך , משתתף נוסף , ומ זהה . " ר נתניהו הציג פני מר טראמפ גברת קלינטון את עמדה של ישראל נושא אזורי ורים ביטחון ישראל מאמץ להשיג שלום יציבות אזור . ר הודה למר טראמפ גברת קלינטון על ידידות תמיכה ישראל " צד השני , מצופה תקופה קמפיין חירות , קימץ פחה פרט . ני מועמד פרסם הצהרה מפורטות , מפארות את מחויבות יחס הברית , תמיכה הסכם הסיוע ביטחוני עשור הקרוב , שאיפה הילחם יחדיו טרור מניעה איראן גרעיני . עד כאן דומה . טראמפ החל אמנם את הקמפיין הצהרה ניטראלי גב הסכסוך , אך שבוע אחרון הוא הולך תיישר ימין עמדה אפילו יותר ממשלה ירושלים . תום פגישה עם נתניהו חרר מועמד רפובליקני הודעה רוויית הצהרה ניצי , מפארת את ירושלים מאוחד בירה נצח של העם היהודי מטילה את אחרית קיפאון מדיני על פלסטיני צריך להכיר מדינה ישראל עם היהודי . הצהרה של טראמפ , פיה " הסיוע ביטחוני מוענק ישראל שיתוף הפעולה תחום ההגנה מפנה טיל הן השקעה מצוין ארץ הברית " ראוי ציון . זאת נוכח העובדה עבר טראמפ קרא ישראל החזיר את כספי הסיוע היא מקבל . הודעה של מטה קלינטון , מאידך , שמה דגש על צורך קידום פתרון שתי מדינה , אשרר את התנגדות כל ניסיון חיצוני כף הסדר , " כולל על יד מועצה הביטחון " התחייבות של קלינטון בעל חשיבות יתרה בחינה של נתניהו , על רקע חשש ירושלים מהלך אחרון של משל אובמה נושא לפני תום כהונה . נתניהו לא העלה את הנושא בכלל פגישה עם אובמה , אבל פגישה עם מועמד " אל נובמבר " מתייחסת תקופה ביניים שבי היוודע תוצאה הבחירות נשיאות כניסה הנשיא החדש בית לבן ינואר בהחלט תפס מקום כובד . הצהרה של קלינטון , מאשררת אמירה דומה בנאום פני שדולה אייפאק מוקדם יותר השנה , כמו גם ההתנגדות גוברת של טראמפ בכלל פתרון שתי מדינה , הן אולי תוצר משמעותי ביותר של ביקור בחינה נתניהו . מי מדקדק וב זאת מחפש את ההבדלים קטנוני פגישה עם טראמפ נמשך שעה דקה , פגישה עם קלינטון קרוב שעה . בעוד טראמפ התרברב כל אמצעי מד החברתית פגישה עם נתניהו , עמוד הטוויטר אינסטגרם של קלינטון אין לה זכר . קלינטון מנוסה ניסח הודעה שהייה יכול קלות צאת בית לבן , שון הודעה של טראמפ נשיאותי קצת פחה .
