שנה אחור : היכן היה אז , המנהיגים של היום ? ה ארגן חבילה חייל חזית , שר אושפז ין פצוע קרב ראש ממשלה לחם שתי חזית . רגע מכוננים ממלחמת כיפור השפיע על דור המנהיגים של שנה חלף אז מלחמה יום כיפור , נערים חייל של אוקטובר הם דור ההנהגה הנוכחי של מדינה ישראל . האזעקה פילח את דממה היום קדוש , חייל חילק צווי גיוס מתפלל בית הכנסה הקרבה הקש סיני רמה גולן , הם זיכרון מלווה עד היום את ראשי המדינה . וואלה ! חדש חוזר אותו רגע מ היא " עיצב את חיים של ראש ממשלה , שר חברי נס השפיע רב על דרך מקצועי , אך בעיקר האישית . עם מלחמה חזר ראש ממשלה נימין נתניהו ארצה אמצע לימודים אוניברסיטה ארץ הברית . " מטוס היה מלא סטודנט ישראלי עזב כל כדי הילחם שדה קרב מען הגנה של מדינה ישראל " סיפר נתניהו , " רב הם נפל קרב " מקביל , השתתף קרב ני אח יוני אח המבוגר עידו אח צעיר , גם הם לוחם סיירת מ . יוני חייל חיסל כוח קומנדו סורי נשלח לפשוט על מפקדת צ רמה גולן . אחר כך ניווט כוח ארטילרי של צ מעמקי שטח סורי את דמשק . נוסף , השתתף קרב מאסף עם כוח סורי נס חרמון . יוני קיבל את עיטור מופת על פיקוד מבצע של יוסי מאחורי קו האויב . עם הגעה , הצטרף נתניהו כוח מילוא של לוחם סיירת מ ירד דרומה גזרת תעלה כדי הגן מפנה התקפה של קומנדו מצרי על ותינו . המשך , עלה ראש ממשלה גזרה צפוני פקד על מבצע מיוחד שלא נחשף עד היום . שנה אחר מלחמה יום כיפור ביקש ראש ממשלה נתניהו להעלות מה כתב אח יוני נהרג מהלך פיקוד על מבצע חטוף מאנטבה אוגנדה ימים ספור אחר מלחמה , דבר שמור עם תמיד ייחוד ביום כיפור : " כמה חזק אומה כמה גדול היא רגע שבר . קשה לתאר איך נקפצות האצבעות אגרוף פלדה ים פרעות " מכתב אחר כתב : " אין אני ספק יכולת כוח רצון הישאר כאן במשך זמן רב כל כך עד נהפוך עובדה מוגמר " " כדי שמר על בית לן , מדינה ישראל , עלה להיות מוכן תמיד לפעול עוצמה נגד אלה מנסים עקור א כאן " ציין ראש ממשלה . " אז מתמיד היה יכולת נחישות הגן על עצם מפתח קיום עתיד . עוצמה היום גדול אין שיעור משהייתה ביום כיפור הוא , אבל אנו שואבים השראה גבורה רוח גילה העם לוחם רגע מבחן . תמיד נזכור את אלה מסר נפש מלחמה הקשה כדי נוכל חיות ארץ " שר הביטחון נזכר מתח אחז בו זמן קצר לאחר פרצה מלחמה . " ככל נקף שעה מלחמה , דיווח לא פסקו להגיע " הוא ספר . " ישבן אז , קבוצה של חייל מילוא חטיבה , בבסיס סירקין , ממתינים שליחה חזית . אלה היה רגע קשה , שורה איוב שדה קרב אל מול חוסר און המוחלט . ידיעה על נפילה מוצב רמה גולן תעלה , על וגים פצוע , על חייל נפלה שבי , אל מול תסכול הרב הצטבר , כן א עדיין לא הפעיל " הוא מתאר גם את תחושה הרעה שרר עם . " הרגשה היה של רעם ביום בהיר . תחושה היה מדינה ישראל נמצא על סף חורבן בית ישי . אם וים נפרץ מצב הוא עד כדי כך גרוע , מי יודע מה יהיה הלאה . תום כמה ימי המתנה , נשלח סיני . משימה ראשון הוטל עלה היה משימה אבטחה נגד קומנדו מצרי נחת חצי האי , אחר כך צלח את תעלה . היה אמור לנוע כיוון איסמעיליה , אך שלב מסוים הסיט א עיר , שם נלחם עד הפסקה האש " יעלון משחזר כיצד מורל עם היה ירוד . " כבר אז , רגע קרב הקש , הבין אני חייב לעשות מעשה . אובדן אמון הנהגה צבאי מדיני היה בד . ידע שאם לא אחזור סייע שיקום הצבא , קצין זוטר לו שנה או תיים , לא אהיה שלם עם עצמי . לא תכנן הישאר צבא קבע . מלכתחילה , עוד חייל שירות חובה ראה את ייעודי לא קורס קצין . היה גם צבא , אך האמין עקב ניצחון גדול של צ מלחמה שת ימים , אפשר לנתב את שאיפה מקום אחר , לא שירות צבאי ארוך . מלחמה יום כיפור שינה את מסלול חיי , כמו גם את השקפה על דרך ההתנהלות של פקד פקוד , של דרג כיר דרג זוטר " " ערב מינוי ראש א קרא את דוח עד אגרנט פגש את גיבור . מפגש למדתי את גודל טרגדיה . בכיר אגף מודיעין אותו שנה היה פירוש אנשים מצוינים . הם שגה כי השתלט עלה , אך לא רק עלה , יהירות , שאננות חוסר צניעות . תחושה אפסי הפך חלק בלתי נפרד מדרך עבודה . לא היה מקום דעה אחר , השקפה נוסף , מחשבה שונה . קרוב לוודאי זהו תמרור האזהרה משמעותי ביותר ניצב פני כל מקבל החלטה , כך אני דורש אורך שנה עצמו גם כפוף אני , מד וש מד : אפשר פקוד כפוף ל לומר את אשר על ליבם דעה , גם אם השקפה אינה תאומת את השקפה . הפגין מנהיגות משתפת מנע עריצות מחשבתית . הבין כי חובה האזין מגוון דעה , כי תבונה חוכמה אינו נמצא רק מקום אחד " בלי צל של ספק , קובע יעלון כי מלחמה יום כיפור היא האירוע מכונן של חיים . " אלמלא מלחמה הזה , סביר היה היום מסלול מקצועי אישי אחר לחלוטין זה אני נמצא בו היום , אך איני מצטער רגע על דרך אליה פנה בגלל מלחמה היא " ביום כיפור של שנה היה שרת משפט נערה בת . " בוש בגד חג לבנים צעד , מדי שנה , עבר בית הכנסה אז שמע פעם הראשונה את מיל " היא משחזרת . " מיל הדהדו , טלטל , עברו ין כולם . יום קדוש ושלו הפך מהירות ליום של תחושה קשה , של חשש קיומי אמיתי , דאגה חוסר און , את זאת ראה על פנים של הורה " היא נזכר שמועה החל לעבור פה אוזן . " סיפור החל זרם ינו גרם כול תחושה חוסר וודאות . הם תואמים את מציאות ? מוגזם ? רק בגרות הבין את המשמעות אמיתי של מה שמע אז . זיכרון אחד צרב אני היטב תודעה , בולט כולם . אני זוכרת את מבט חיוור על פניה של אם מבט זכר אז , אבל הבין רק כשהפכתי אני עצמו אם . זה היה מבט של אם פותחת את דלת בית מולה אדם חאקי . איש לא בא ספר על בן , אח , אלא בקש את נו נעדר . הוא דפק על דלת בתים של כל חבר כדי ברר אולי אם שמע משהו . מלחמה הסתיים . איש חאקי הפך לאב כול , דורון נו השני בן כיתתי , הפך אח כול " שר מדע טכנולוגיה , ראש שירות הביטחון כללי לשעבר , היה ירושלים עת פרצה מלחמה . הוא התגורר שם עם משפחה שירת ראש מחלקה סיכול פעילות חבלן עוין פ מרחב יהודה שומרון ירושלים . " איש מילוא של שירות הביטחון כללי מיהר למטה המרחב , שב התקבץ כל עובד עובד " הוא מתאר . " על פני כולם ארשת של אי ודאות דאגה , על גבול תדהמה . למטה המרחב החל הגיע ידיעה ראשון על מתרחש חזית צפוני חזית דרומי . אוזן שמע אך ראש עיקר לב סירב האמין . פקד המרחב אותו עת , יהודה ארבל פקד , נחוש , איש מודיעין מנוסה שמורגל הפתעה מצב של אי ודאות , כינס את חבורה מפקד , דיווח לן על מידע ראשוני היה רשות " " בסיכום דבר קבע ארבל כי עבודה המרחב תחולק שלוש רמה " ספר פרי . " תחילה , הוא הורה על פתיחה כל מטה אזורי על מנה אסף מודיעין אודות תכנון פיגוע כוונת חבל שיירה או בסיס של צ , כניסה מצב חירום דרגה מירבית . שלב השני , הורה ארבל יצור קשר עם כל מוסד אזרחי אמור היה טפל תושב עורף עירייה וועדים מגזר היהודי מגזר ערבי . שלב שלישי , עוד לפני תמונה המצב היה ברור , הוחלט כי קצין קישור של ש יוצבו כל יחידי צבאי על מנה לתאם את העבודה עם צ , סייע כל דרוש לפעול אור כל תרחיש מתפתח תוצאה מקרבות בלימה " דבר פרי , " ההפתעה שהייה מנה חלק של תושב ישראל היה דומה לזו שררה שטח . כן , ים הראשונים שטח היה שקט אופן יחסי , אך מוצאי בת מורל של תושב יהודה ושומרון עלה התמדה תושב יצאו רחוב ייחל מפלה ישראלי . אנשי יחידי עליה פיקד החל לפעול סיכול פיגוע מתוכנן , הפרה סדר ניסיון פגיע בסיס של צ שיירה הסיע חייל טנק חזית צפון דרום . טקסט קצר להכיל את הרגשה פעולה סיכל ביצע מהלך כל ימי הלחים הקש " " מי לחם מלחמה שת ימים , קרב דמים רמה גולן , סירב האמין בשעות ראשון אנו עיצום של הפתעה מודיעיני חסרת תקדים . אני זוכר את עצמי מנסה למצוא הסבר התיימר ענות ע על השאלה איך זה קר . לצערי , התפכחות היה מהיר כואב , מכל היבט אפשר לעלות על דעת . ראש ראשון היבט לאומי , דגש על העובדה היה כבר איש מודיעין תיק יחסי . אך לא פחה כך , היבט אישי , כן לא היה שעה או יום מהלך לא קיבל ידיעה על חבר איבד או אלה הוכרז נעדר . לעולם לא אשכח את בתי קבר ארעי את שיירה ארון , עטוף דגל לאום , דרך קבורה כל מקום . במשך בועה רב לא היה יום אחד שלא נכח שתיים או שלוש לווה צבאי של חבר קרוב אשר לווה א עד היום . פרספקטיבה של שנה אחור דבר הלכו התעמעם , אך תחושה חורבן חוסר און , היה קשה נשא . רק התגייסות הלאומית , כמו שעם ישראל יודע התגייס מצב אלה , קלה מעט את תחושה הקשה " אם את מ המשטרה תפס מלחמה בית הכנסה . " אז גיל ילדות אני נוהג פקוד את בית הכנסה ביום כיפור " הוא ספר . " זמן פרצה מלחמה , מתפלל רב קיבל צווי גיוס אני חבר , תלמיד כיתה אותו ימים , הביט מבוגר הבחין אי ודאות שבה היה שרוי . נער זה הותיר ב רושם עז , טראומתי . אי וודאות תחושה פחד אחזו גם ב . יוון אב גויס מלחמה , אני משפחתי נסע בית סב וסבתי יפו " " ים מלחמה חשב כיצד אני יכול תרום לאומי , יחד עם כמה נערים נוסף עברנו בית בית , אסף תרומה הכין חבילה חייל חזית . פעל מסגרת עמותה אנשיה ארגן חבילה חייל , אני הופקד על אזור דרום תל אביב , יפה , חולון בת ים . כל מי שפן אל הכניס א ביית תרם . תור ילד , אתה מעצים את דבר אתה חווה עד היום אני זוכר את תחושה הזה של אי וודאות . יום הראשונים מלחמה לא ידענו מה קורה , זו תחושה עד היום יושבת אני ראש . יחד עם זאת , ככל ארך הקרבה , החל הגיע דיווח על הישגי צ שדה קרב . התעורר תחושה של ניצחון , החל הגיע שורה טוב . כמה שנה לאחר כן , כאשר התגייס חטיבה הצנחנים , תחושה אותו מלחמה ליוו א כל יום . רצון הוביל א התגייס שירות קרבי לתרום תחושה שליחות קינן ב , מלוות א עד היום " שר אנרגיה מים , נער באר שב , יצא גם הוא בית הכנסה הבין משהו גדול קורה . " אותו ערב , כאשר יצאנו בית נס , ראה שיירה ארוך של משאית נוסעת כביש ראשי עיר עם טנק . גר אותו תקופה שכונה גבל כביש ראשי מוביל דרומה , אני זוכר את אותו שיירה היטב יוון זה היה חזה מוזר . יום כיפור , אף מכונית לא נע רחוב אך שיירה ארוך עם טנק עוברת סמוך שכונה . למחרת , בוקר יום כיפור , כבר היה ברור כי משהו קורה אך עדיין לא ידענו וודאות מה מתרחש . אני זוכר כיצד אמצע תפילה נכנס אנשים בית נס , לחש על אוזן של חלק מתפלל אל יצאו ותיהם כבר לא שב תפילה . שעה , מיד לאחר תפילה מוסף של יום כיפור , פילח אזעקה את אוויר קולה מייבב נשמע מרחק . אז כבר היה ברור פרצה מלחמה " " יצאנו מיד בית הכנסה , עושים דרך עבר בית , מכונית החל לנוע כביש . אני זוכר את בן דוד , חייל סדיר , עולה על ווספה שלו נוסע בסיס פיקוד דרום . ים אחרי , ימי מלחמה , התנדב תחנה דלק כביש יציאה דרומי עיר , שם חילק חייל היה דרך חזית סיני משקה קל , עוגה סיגריה . לילה אז היה חשוך של האפלה , אך אנחנו ראה נגד עין רק את אפשרות סייע ככל ניתן חייל צ היה דרך שדה קרב בדרום . בלילה לאחר הפסקה האש , אוקטובר , אני חבר טייל עיר נפצע . במשך שלושה בועה אושפז בית החולים , צד חייל צ נפגע קרב . שום מה סבר רב כי גם אני חייל נפצע מלחמה חבילה שי לא פסקו הגיע חדר . אפילו בדרן , דני קיי , הגיע בקר א . אך רק אותו ימים שב שהה צד חייל , הבין את גודל הבעיה שבה היה נתן ישראל מלחמה את ההצלחה הירואית סיום , כ נמצא עמוק מצרי , בואכה קהיר . שנה לאחר מלחמה הקשה היא , אפשר הבין מה חשיבות עומק אסטרטגי מניעה מלחמה , מה חשיבות הסכם שלום שכל מהות לש סוף מלחמה מעגל דמים " זיכרון מלחמה נצרב ראש של חבר הכנסה עבודה עד היום . " כל כך הרבה שנה חלף ין גיל גיל , אך זיכרון טרי , כאילו אירוע מתרחשים רגע זה ממש . במה התחיל ? שמועה , מכונית החל על הכבישים או רגע נחרת זיכרון עולם עד , שב הגיע בית הכנסה של משפחתי צמד חייל שאל היכן גר עזרא כבל , אח של אבי ? רגע הזה לא יישכח , כי הזמן היה זמן קריאה ספר יון בית הכנסה , השעה ערך , והם יקשו ממנו סיים , גשת ביית סמוך להתלוות יהם יחידה . זה רגע שב הבין כי דובר עניין רציני , יוון שהם השתמש צמד מיל הגדולות , . זה לא זמן משחק ואין זמן השתהות עד כדי הפסקה קריאה של הפטיר יון " " רגע כונן נוסף אותו מלחמה , רגע מלווה א עד היום חיי בוגר , הוא המשפט מהדהד ראש של ה דוד אלעזר נשבור הם את . עד היום אני זוכר את קול סמכותי אך סדוק של דד , בקע טלוויזיה , זה כל מה אני זוכר דבר . אותו רגע הבין , נער צעיר , שאנחנו באמת בצורת גדול . מדוע נדרש איום הזה ? מה צריך הודיע על כך ? הרי תמיד הודיע רק אחרי שבר הם את עצמות , מה השתנה הפעם ? הסתבר אכן הפעם המצב היה שונה , כי עת הודעה של ה , האויב היה זה שבר לן את עצמות " שנה מלחמה יום כיפור כותרת נ תשחצים : הכרזה של מלחמה יום כ נחשף : טוסים הציל את גולדה לאגרנט נחשף : את שנה אחור : היכן היה אז , המנהיגים של היום ? ה ארגן חבילה חייל חזית , שר אושפז ין פצוע קרב ראש ממשלה לחם שתי חזית . רגע מכוננים ממלחמת כיפור השפיע על דור המנהיגים של
