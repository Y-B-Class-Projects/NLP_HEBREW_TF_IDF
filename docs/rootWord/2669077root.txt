שר אישר : אסיר ישוחררו , חלק אנשי ח פורסם אתר שירות בתי סוהר . אסיר יועברו רצועה עז היתר שטח הגדה . אסיר ישתחררו ככל הנראה בוקר יום רביעי . אבו מאזן : אסיר לא יגורש : דיון סוער עתירה נגד שחרור אסיר , אתמול קבוצה אסיר ראשון בדרך רמאללה : צוות שר מצומצם נושא שחרור אסיר פלסטיני גיבש אמש ראשון את רשימה אסיר פלסטיני ישוחררו ים קרוב . אסיר , הם גם אנשי חמאס , ויים הגיע רמאללה כל מאוחר ביום רביעי בוקר . דובר קבוצה ראשון תוך אסיר פלסטיני תיק ישראל התחייב שחרר ארבע פעימה אורך משא ומ . רשימה אסיר מלא פורסם אתר שירות בתי סוהר , לאחר הודעה נמסר משפחה שכול ביקש קבל הודעה ראש . רשימה אסיר רצועה עז עוד הגדה מערבי , נחשב אסיר פחה סוכן בין אסיר וותיקים . ין משוחררים : הורשע רצח שאר סיוע או ניסיון רצח . הודעה אתר נכתב כי החלטה צוות התקבל לאחר נציג רשות שיש הן נגיעה דבר שירות בטחון כללי , שרות בתי סוהר , פרקליטות צבאי , משרד משפט בחן את התאמה של אסיר שחרור . " יצויין כי ההחלטה דבר שחרור של אסיר כפף שיקול דעת הגורמים מוסמך ולהחלטה השגה יוגשו " נוסף , נכתב אתר , ניתן קבל ידע על אסיר על תהליך שחרור מוקד טלפוני הפועל מחלק חנינה משרד משפט ין שעה עד מועד שחרור . אסיר ישוחררו פחה שעה אחרי פרסום רשימה . אסיר הם תושב עז ול יועברו רצועה , בעוד היתר יועברו שטח יהודה ושומרון . מונה בין אסיר אמור היה השתחרר בתי הכלא שלוש שנה קרוב , ניים הם חצי השנה קרוב . בדיון וועדת שר הודגש שאם אחד משוחררים ישוב פעילות עוין נגד מדינה ישראל הוא יוחזר המשיך ריצוי תקופה מאסר . ארגון אלמגור פרסם הודעה על רקע פרסום רשימה אסיר : " זהו יום חג ארגון טרור פלסטיני יום עצוב משפחה שכול גם חבר ישראל " נכתב הודעה מטעם הארגון . " תווך נמצא פוליטיקאי ישראלי נכנע אל מול טרור , מכס על כך מסכת של פעלול י . צריך לומר יושר ארגוני טרור ניצח את נתניהו ממשלה . אלמגור מעתה יהפוך שמ כדי היאבק נגד טרור כדי הביא הענשה טרוריסט " צוות שר כולל את ראש ממשלה נימין נתניהו , שר הביטחון שה בוגי יעלון , שרת משפט ציפי בן , שר ביטחון פנים יצחק שר מדע טכנולוגיה יעקב פרי . יוון ראש ממשלה נתניהו שוחרר רק אמש בית החולים אילו שר נמצא עת ב , את רשימה אסיר ישוחררו שבוע גיבש אך רק שר יעלון , בן פרי , יחד עם גורם מקצוע מערכת הביטחון . גורם מדיני אמר אמש וואלה ! חדש כי רשימה צפוי כלל את אסיר פחה " יוקרתי " מנקודת מבט פלסטיני , בין אותו אסיר תיק הם אסיר בוגר , חולים או אלה עומדים השתחרר בקרוב כל קרה . שלב זה , נראה כי את אסיר ה " ביותר תשחרר ישראל רק שתי פעימה האחרונות , כך פלסטיני לא ימהרו את שיחה , אפשרות רגע יממה האחרונה נראה היה עומדת על פרק , על רקע מכרז הבנייה חדש התנחלות , נחשף הבוקר וואלה ! חדש . י הרשות פלסטיני אבו מאזן אמר אמש כי אף אסיר בין אסיר ישראל צפוי שחרר בתי הכלא לה , לא יגורש מ " שלו . מהלך פגישה עם שר עניין אסיר רשות עיסא קראקע , אישר אבו מאזן כי אסיר ישוחררו ארבע פעימה . מוקדם יותר אמר ראש צוות משא ומ פלסטיני סאיב עריקאת כי הבנייה התנחלות פוגע משא ומ , הוסיף כי " מי אחראי כך , נס א עזב את שולחן משא ומ " עריקאת תקף את הנחיה של ראש ממשלה נתניהו גרש אסיר מערכת הביטחון תגדיר מסוכן רצועה עז , הצהיר : " כל אסיר ישובו כפר , אף אחד לא יגורש " גם סגן של עריקאת צוות משא ומ פלסטיני , מוחמד שתיה , תקף את שתי ההחלטות , אמר כי מתקבל רושם ש אינה רציני עניין משא ומ " תזמון מעט מושלם , הגיע אמש אזור מי אמור להרגיע את דים חדש את אמון ב שליח מדיני אמריקני מרטין אינדיק סגן , פרנק לוונסטיין , נפגש אתמול רמאללה עם אבו מאזן , ירושלים עם נשיא המדינה מעון פרס . תוכן שיחה של אינדיק עם פרס אבה מאזן לא פורסם , אך גורם שמעורה פרט אמר כי פרסום מן יממה האחרונה " הוזכר יותר פעם " על רקע אווירה עכר הזה אמור צוות משא ומ חדש את שיחה ישיר ביום רביעי הקרוב , סבב שיחה יחל ירושלים יעבור לקראת סוף שבוע יריחו . את צד ישראלי תייצג שרת משפט ציפי בן יחד עם שליח מדיני יצחק מ , את צד פלסטיני ייצג סאיב עריקאת יחד עם סגן מוחמד שתיה . גורם שמעורה משא ומ אמר וואלה ! חדש כי ניגוד סבב שיחה הקודם וושינגטון , היה עיקר " טכני טקסי " סבב הקרוב יעסוק סוגיה ליבה מדיניות . סיום של סבב שיחה , שליח אינדיק אמור להתמקם אופן קבוע מזרח תיכון להמשיך קיים פגישה עבודה נפרד עם דים . עדכון ראשון : ני : ידיעה השתתף אבי י תהליך שלום כותרת א נערך שחרור אסיר ה ראשון : משרד שיכון אישר בנייה דירה רק ב סוער שחרור אסיר : פעם משתחרר ? כתב אמיר תיבון שר אישר : אסיר ישוחררו , חלק אנשי ח פורסם אתר שירות בתי סוהר . אסיר יועברו רצועה עז היתר שטח הגדה . אסיר ישתחררו ככל הנראה בוקר יום רביעי . אבו מאזן : אסיר לא יגורש
