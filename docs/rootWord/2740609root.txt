ה עוד כישלון מדיניות של א כימי סוריה , משבר אוקראינה סל הריק של הישגי ממשל אמריקני בעולם מצטרף עת תהליך שלום , שב הושקע רב . הטלה אחרית על דים לא חוסכת נשיא את ה עיתונאי נערך אתמול שישי בית הכחול סיאול , מעונה רשמי של נשיאה קוריאה דרומי , פארק , דיבר נשיא ארץ הברית , ברק אובמה , על תהליך שלום השליך ין רגע פח את כל לוח זמן תאריך יעד עלה עמל צוות תשעה חודש אחרון . " זו עי נמשך , , שנה " אמר . " לא ציפה נפתור את זה מהלך משא ומ נמשך שישה או תשעה חודש " לפי התבטאות עבר של שר אמריקני קרי , הוא דווקא כן האמין זה אפשרי , אבל מה להיות קטנוני ? " זה עדיין אינטרס אמריקני , ישראלי פלסטיני פתר את הסכסוך הזה " אמר אובמה . " עד כה , פחה ראה איזה תנועה שני דים הודות משבר ממושך הזה דורש פתרון . מה שלא ראה , מען אמת , זה את רצון פוליטי קבל החלטה קשה . זה נכון גב ני דים " תהליך שלום כל ידיעה ה ב : ישראל הציע רשות לבן שטח יודע : ממשלה האחדה פלסטיני תכיר ב הודות אבו מאזן , נתניהו החליט נתק מגע . פרשנות " החלטה של אבו מאזן קיים שיחה עם חמאס היא רק צעד אחד סדרה של החלטה קיבל גם ישראלי גם פלסטיני , אינו עוזרות לנסות פתר את משבר " הוסיף אובמה . " אני לא מתנצל על תמיכה מאמץ בלתי נלאה של קרי , למרות סיכוי נמוך , המשיך לנסות גשר ין דים " אף אחד מסיבה עיתונאי בית הכחול לא ביקש נשיא אמריקני התנצל הוא אמר את זה על דעת עצמו , הוא מודע היטב ביקורת אף גיחוך עורר אחרון מאמץ של קרי נהל שיחה על המשך שיחה , לבסוף נחל כישלון " ייתכן מגיעה נקודה שבה צריך פשוט לקח הפוגה שני דים יוכלו בחון את אלטרנטיבה " הרהר אובמה קול . " אף אחד לא הציע אני תסריט רציני , מסגרת ישראלי פלסטיני אינו חותמים על הסכם שלום , אך כל זאת יש לן דין יהודי בטוח דמוקרטי , פלסטיני יש דין משל . יכול לעשות הצגה , עמדה קיצוני , אך אופן מציאותי , יש אפשרות אחת שני דים יתכנסו יחד יעשו פשרה פוליטי קשה כדי הבטיח את עתיד דור הבאים . טרם ראה אותו מגיעים שלב הזה , אולם נמשיך עודד אותו " " האם אני מצפה הם הגיע שלב הזה שבוע בא , חודש בא או אפילו ששת חודש הבאים ? לא . האם נמשיך לנסות להשתדל הציע גישה קונסטרוקטיבי יכולת הוביל אותו קדימה לעשות את צעד האלה ? בהחלט . אני לא מתנצל על כך . זה דבר נכון לעשות . זה אינטרס לאומי של אמריקה , גם האינטרסים של אזור ושל ישראל " נוכח ההשקעה של משל אובמה ניסיון קדם את תהליך שלום , אחרית דים עצמו כישלון הנוכחי אינה חוסכת נשיא את מחיר מבוכה . כהונה נוכחי של אובמה סל הישגי דיניות שלו נותר די ריק מרבי נשק כימי אמנם פונה מסור , אך שטר אסד ממשיך על פי דיווח שונה את תושב חלב חבית מלאה כלור , שיחה עם איראן על תכנית גרעין לה טרם הוכתר הצלחה אסר שכח את משבר אוקראינה . אותו מסיבה עיתונאי סיאול , אמר אובמה נשיא רוסי , ולדימיר פוטין , הוא אמנם " לא אדם טיפש " הוא בין מה מחיר רוסי תשלם על בחיר אופציה צבאי אוקראינה אולם הודה כי עיצומים הוטל על מוסקבה לא יפתרו הכרח את משבר או יגרמו פוטין שנה את דעה עניין . " זה רק יכול הפך את ההחלטה קשה יותר ע " אמר אובמה . הסכסוך מזרח תיכון זה לא משהו שאובמה , ככל מאס בו , יוכל להתנער ממנו . אולם , הוא בהחלט יוכל הוריד את הילוך להתמקד שפע אתגר אחר עומדים פנים . בינתיים , משל אובמה אינו ממהר שבר את כלי להכריז על מות תהליך , אלא היפך בחינה ממשל , עתה נמצא כדור מגרש פלסטיני . נוכח הכרזה קודמות על האחדה , אמריקני ספקן גב סיכוי הצלחה פיוס ין פלג פלסטיני . אם אכן תוקם ממשלה אחדות פלסטיני , תכיר ישראל הסכם קודם א , ייתכן יופעל על ישראל על נימין נתניהו , אף מסתכן כך יופנו אצבע מאשימות לפי קואליציה שלו , שלא כל חבר קשובים למאווייהם לאומי של שכן עבר גדר הביטחון . למרות אכזבה , אמריקני היה רחוק מ את דלת אבו מאזן . שליח אמריקני מרטין אינדיק נפגש עם , קרי אירח ביום חמישי בערב משרד ארוחה ערב חגיגי מטרה משך השקעה כלכלה פלסטיני . דובר משרד , סאקי , כינה את קיפאון הנוכחי " רגע עבר הוא חלק תהליך " היא אמרה כי " תהליך שלום רדיפה אחר שלום הסכם סופי זה לא משהו שאנחנו מוותרים עלה , אולם אינו יכול כף על דים לעשות צעד אינו רוצים לעשות " קרי , הדגיש סאקי , עדיין חושב הסכם שלום הוא " אינו משימה בלתי אפשרי " האם הקים ממשלה פלסטיני חדש , עתיד כלל את ארגון חמאס , שאותו א מגדירה ארגון טרור , תביא הכרח הפסקה הסיוע אמריקני רשות פלסטיני ? כמה חברי קונגרס כבר איים לנקוט צעד סוג זה , כמה ארגון יהודי אמריקני משפיעים , כולל שדולה א , הזהיר פיוס פלסטיני פוגע סיכוי קדם את שלום . יחד עם זאת , נשמע גם קול אחר . ראש ה לשעבר עם איילון , נמנה על מייסד הארגון " עתיד חול בן " מקדם את פתרון שתי מדינה , כתב ב יורק טיימס " שבוע כי פיוס ין פלג פלסטיני " הוא צעד משמעותי לקראת הסכם כולל ין ישראל פלסטיני " נשיאה הארגון " אמריקני מען שלום עכשיו " דברה דה אני , הגדיר את פיוס פלסטיני " חדש טוב , שגם משל אובמה גם ממשלה ישראל צריך לראות אותו ככאלה פרטנר פלסטיני , מייצג את כל העם פלסטיני , מסוגל קבל החלטה קשה יד שולחן משא ומ , לספק תוצאה כשמגיע הזמן ליישם את הסכם שלום " ה עוד כישלון מדיניות של א כימי סוריה , משבר אוקראינה סל הריק של הישגי ממשל אמריקני בעולם מצטרף עת תהליך שלום , שב הושקע רב . הטלה אחרית על דים לא חוסכת נשיא את מבוכה
