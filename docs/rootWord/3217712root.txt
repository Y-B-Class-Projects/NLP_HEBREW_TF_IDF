טורף נתניהו : שיא מנדט ליכוד , עם כוכבי מבהילה ראש ה אימה של ליברמן אורלי לוי , אופטימיות של לפיד אפקט חיסול של גדעון סער : יום פריימריס מוצלח של מפלגה השלטון נותן ותיו גם סקר מעודכן של וואלה ! לאחר פריימריס ליכוד , נראה נבחר בחר מפלגה השלטון מחזקת את עמדה ציבור ומביאה אותו שיא של מנדט אז פתיחה מערכת הבחירות : סקר וואלה ! מכון פאנלס פוליטיקס ראשות מנחם זר עלה כי אם הבחירות היה נערך היום , ליכוד היה מקבל מנדט מנדט יותר סקר הקודם נערך שבוע שעבר . עם זאת , התחזקות המפלגה גורמת כך שת פלגה אחר גוש ימין פוטנציאלי ישראל יתנו גשר של ח אורלי יורד מתחת אחוז חסימה , לעומת מנדט שת מפלגה קיבל סקר קודם שבוע אחרון . שתי פלגה אלה , גם מפלגה ימין קיצוני " עוצמה יהודי " של ברוך מרזל איתמר בן גביר מצוי על סף מעבר של אחוז חסימה עם . כך שמה עולה כי ליכוד אמנם מתחזק , אך כפי נתניהו מזהיר שבוע אחרון , ישן סכנה גדול מנדט ימין יישארו מתחת אחוז חסימה , תרחיש פוגע משמעותי גודל גוש ימין של קואליציה נוכחי מנדט בלבד , לעומת כנסת יוצאת , יחד עם ליברמן פרש רק ישורת האחרונה . חירות סיקור מיוחד וואלה ! : לוקחת את אמונה צעד אחד קדימה " אלונה ברקת מצטרפת מפלגת " ימין ה דיל , תודה רב , תודה גלעד רון : הפתעה נגב ליכוד סקר עולה כי מפלגה " חוסן ישראל " של ה לשעבר בן ממשיך לצבור מומנטום , מגיעה מנדט סקר הקודם . גם " יש עתיד " של יאיר לפיד מתחזקת קלות , מנדט אחד , מנדט . מפלגה ימין יש תזוזה קל של מנדט אחרי הכרזה על בחיר של הרב ראשי לשעבר רפה ראשות בית היהודי , מפלגה מאוחד שלו ושל בצלאל מקבל מנדט לעומת שבוע שעבר . ימין החדש של נפתל בנט איילת שקד נחלש מנדט אחד תוצאה סקר הנוכחי נערך לפני הצטרפות של אלונה ברקת מפלגה . שמאל , לעומת זאת , המצב הרבה פחה אופטימי . התחזקות ממשיך פגוע בעיקר מפלגת העבודה , יורד שפל של מנדט , מה הופך את פריימריס מפלגה שבוע בא קרב אכזרי מיוחד על ספר קטן של קומה ריאלי . מרצ ממשיך לא להרוויח דבר אובדן קול אצל ותה גדול , נותר יציב על מנדט גם היא . מפלגה תנועה של ציפי בן נותר רחוק מאוד אזור אחוז חסימה , עם משיב . כול , לעומת זאת , לא נפגע לא לא נתניהו ממשיך שמר על בסיס יציב של מנדט , כך גם מפלגה חרדי : יהדות תורה עם מנדט ו עם , מפלגה ערבי : רשימה משותף עם מנדט ו של אחמד טיבי עם . סיפור פוליטי לוהט של פריימריס ליכוד היה מבחן קאמבק של גדעון סער מפלגה , צל ניסיון חיסול פוליטי חד , ישיר בוטה כיוון ראש ממשלה . נתניהו מקורב עש גדול כדי מנוע מסער את הבכורה , והוא הגיע מקום בלבד , אך בינתיים הם הצליח סמן אותו יריב פוליטי ספר אחת שלו לתת לו עמדה הובלה משמעותי קרב על ירושה . סקר וואלה ! עולה כי בין כל חברי חמישיה פותחת של ליכוד פריימריס , סער הוא מועמד מועדף הוביל את ליכוד ביום אחרי נתניהו גם ציבור כללי גם קרב מצביע ליכוד . ציבור כללי הוא מוביל פער גדול על פני יולי אדלשטיין , אמנם כבש את המקום ראשון פריימריס אבל רק סמן אותו יורש , ישראל , גלעד ארדן מירי רגב כל אחד . גם מצביע ליכוד סמן את סער מועמד מועדף עלה , אבל פער קטן יותר חבר חמישיה . ליכודניקים השיב סקר אמר כי גלעד ארדן הוא מתאים ביותר הוביל את ליכוד ביום אחרי נתניהו , תומך ב , . דווקא אדלשטיין , זכה מניין קול מתפקד הצביע בחיר המקדימות , לא זוכה תמיכה רב קרב מצביע ליכוד נותנים לו בלבד . נתון ציבור כללי גב סער גם אשר מועמד אחר , התאמה מעוות מידה מה , כי הם משקפים גם הערכה של מי אינו מצביע ליכוד מתנגד נתניהו לפי שר לשעבר , מתויג עת יריב של ראש ממשלה . נתון רלבנטי בחינה המאבק על הובלה ליכוד ביום אחרי נתניהו הוא קרב מצביע ליכוד , שם פער ין סער ארדן קטן יותר . צד השני של מתרס , שם חם של שבוע אחרון הוא של ה לשעבר גבי אשכנזי , מחוזר אינטנסיביות על יד כדי הצטרף אל וליעלון מפלגת גנרל , פועל עצמו ניסיון הביא חיבור ין לפיד , לפי סקר צפוי הביא מהפך . סקר עולה כי ציבור , כלל , מקבל שוויון נפש את אשכנזי מתחלק ין היה רוצים לראות אותו נכנס פוליטיקה , שלא היה רוצים , משיב אמר אין הם דעה . עם זאת , אשכנזי מאוד מאוד רצה אצל מצביע של לפיד תומך רוצים לראות אותו מצטרף לזירה , רק יש עתיד חוסן ישראל השיב שהם לא מעוניינים לראות אותו פוליטיקה . סקר נערך לאחר פרסום רשימה סופי פריימריס ליכוד , קרב פאנל אינטרנטי של משיב המהווים מדגם מייצג של האוכלוסייה היהודית ערבי ישראל . טעות דגימה הינה . סקר נוסף פורסם ערב ב " עולה כי אם הבחירות היה נערך היום , ליכוד היה מקבל מנדט חוסן ישראל מנדט . עוד עולה כי יש עתיד מקבל עשרה מנדט , ימין החדש טרם הודעה על הצטרפות של אלונה ברקת תשעה מנדט , רשימה משותף עם שבע מנדט ו עם שישה . ש , יהדות תורה , בית היהודי מפלגה העבודה מקבלות חמישה מנדט כל אחת בעוד מרצ , ישראל יתנו כול מקבלות ארבעה מנדט . מפלגה גשר של אורלי התנועה של ציפי בן לא עוברה את אחוז חסימה . עוד עלה סקר כי אם לפיד יתאחדו הם יעקפו את ליכוד עם מנדט , אך גוש לא יגדל אחר מרצ תרד מתחת אחוז חסימה . עדכון ראשון : טורף נתניהו : שיא מנדט ליכוד , עם כוכבי מבהילה ראש ה אימה של ליברמן אורלי לוי , אופטימיות של לפיד אפקט חיסול של גדעון סער : יום פריימריס מוצלח של מפלגה השלטון נותן ותיו גם סקר מעודכן של וואלה !
