ח הרצוג : " שש שנה הרעה של נתניהו תם " י העבודה הביע שביעות רצון הליכה בחיר אמר : " ישראל מגיעה הנהגה טוב יותר ראשות " לפיד הודיע הרצוג שלא יכהן ראש אופוזיציה . ח גלאון עקץ : " נתניהו הכריז נכשל , אז תבחרו ב שוב " המערכת פוליטי סוער בעקבות הודעה ראש ממשלה נימין נתניהו על הליכה בחיר . " מדינה ישראל עומדת פני מערכת חירות גורלי . שש שנה הרעה של נתניהו תם " אמר אמש ישי י אופוזיציה מפלגה העבודה , ח יצחק הרצוג . " ישראלי מגיעה הנהגה אחר , טוב יותר , מאחדת יותר , חברתי יותר אחראי יותר . גוש ציוני שפוי ישלב יים ראשות מפלגה העבודה יחזיר את תקווה . הבחירות ברור : זה או ליכוד או העבודה . אנחנו ננצח " אחרי פוטר על יד נתניהו יחד עם שר , שוחח שר האוצר יוצא לפיד עם ח הרצוג . לפיד אמר עמית כי הוא לא מתכוון בקש עמוד ראשות אופוזיציה על אף שיש עתיד ממשלה נתניהו על אף היא סיעה אופוזיציה גדול ביותר עד הבחירות , תקיימו כנראה הקרוב . כמו כן , שני לא דיבר על ריצה משותף רשימה אחת בחיר . ישראל בדרך בחיר עוד כ הודיע על חירות : " לפיד ניסה לעשות פ יש עתיד התפטר : " נתניהו התנהג ב המתנה : מסלול מפותל בדרך אל קלפי " נאום של נתניהו מסתיים בדיוק כמו תקופה כהונה : עיוורון ניתוק מוחלט ין מיל גבוה מציאות עגום " האשים י מרצ זהבה גלאון . " מי פספס את מסיבה עיתונאי של נתניהו , הנה סיכום מסר שלו : נכשל תבחרו ב שוב " גלאון הוסיף : " חבל לבנה לפיד חייך נתניהו יפטר אותו , הם היה צריך התפטר מזמן לא להיות ה של ממשלה הזה " המשך הודיע כי אין כוונת , שום אפשרות , חבר ממשלה תורכב על יד נימין נתניהו . " מרצ לא תשב ממשלה ראשות נתניהו . נקודה . מרצ לא תמכור את קול ש לא תבייש א . מרצ גדול תהיה שותף אך רק ממשלה אלטרנטיבי לא שידור חוזר של זו נוכחי " חברה סיעה ניצן הוסיף : " טוב שהם הולכים הביתה . כולם . כי ממשלה הזה נולד חטא , עשה נזק בלתי פוסק , עכשיו היא מגיעה אל סופה ראוי : פח . היא נולד חטא כי אסור היה לפיד ול חבר ימין קיצוני . מי בחר הם לא רץ את בנט ליברמן בטח לא רץ את ביב . מנדט של לפיד ול נתן ימין קיצוני ממשלה , את תוצאה כול הרגיש : מלחמה מיותר תבערה כללי , קפיצה ביוקר מחיה , קיצוץ שירות ציבור , בידוד בינלאומי , גל של גזענות . . כמה הם הספיק שנה חצי , מדהים " י העבודה לשעבר ח שלי יצא אף היא נגד דבר נתניהו אמרה כי " הוא דיבר כאילו לא הוא היה ראש ממשלה תיים האחרונות . צפה מופע של תשבחות עצמי שב דבר נתניהו על ממשלה הוא עמד ראש , כאילו לא הוא היה ראש ממשלה , זו התנערות מוחלט אחריות . הוא מאשים את כל העולם אשר את עצמו פורס חזון עצוב נטל תקווה מדינה . צודק נתניהו שותף כשל , צודקים גם הם הוא כשל " עמיתה , ח סתיו פיר , אמרה כי אחרי הודעה על חירות המשמעות היא ש ישראל כבר לא תקוע עם ביב " בהתייחסה קמפיין שהשיקה מפלגה עוד קודם פירוק קואליציה . " הגיע הזמן ציבור ישראל יפני את זה יפעל ביחד א כדי החליף סוף סוף את השלטון ישראל . צמד מסוכסכים לפיד נתניהו , התחרה ב שנה חצי האחרונות בעיקר על דבר אחד : מי עושה הכי פחה מען ציבור ישראלי , הכי הרבה שביל מינוף עצמי . מחיר דיור עלה , מחיר שכירות קפץ , שכר נותר קפוא , ה מדיני נכשל , איום ביטחוניים על ישראל תוך שיא , העם מפולג גזענות חוגגת . הבחירות האלה לא רק לא מיותר , הן הכרחי . זו ההזדמנות לן , מחנה דמוקרטי ישראל הובלה מפלגה העבודה , להראות אפשר גם אחר . זה הזמן לן הביא את השינוי ציבור ישראל כל כך מייחל לו " פנייה כתב עמרי נחמיאס : עדכון ראשון : ח הרצוג : " שש שנה הרעה של נתניהו תם " י העבודה הביע שביעות רצון הליכה בחיר אמר : " ישראל מגיעה הנהגה טוב יותר ראשות " לפיד הודיע הרצוג שלא יכהן ראש אופוזיציה . ח גלאון עקץ : " נתניהו הכריז נכשל , אז תבחרו ב שוב "
