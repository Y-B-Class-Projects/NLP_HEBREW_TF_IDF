ר שוחח עם נו של פואד , ח קרא נתן לו ק ממשלה שוחח עם יאיר בן אליעזר , נו של חבר הכנסה מאושפז זה שבוע של דלקת ריאה . ח איוב קרא הגיע בקר , מסר לו קמע דבר הציל את חיים היה קצין תחת פיקוד של פ ממשלה נימין נתניהו שוחח ערב רביעי עם אחד מחמשת בן של ח נימין בן אליעזר , יאיר , אמר לו : " אני מחבק א מאחל ל התאוששות מהיר . אנחנו דואגים אוהבים מאוד את פואד . אני ממתין קוצר רוח כך יפקח את עין אגיע בקר רגע זה יקר . אנחנו מקווים מאמין שיחזור בקרוב חיק משפחה לחיים ציבורי מתפלל שלום " שר הביטחון לשעבר מאושפז בית החולים אסף הרופא ראשון ציון זה שבוע , ימים אחרון החריף מצב . הוא סובל דלקה ריאה , ימים אחרון הינו מורדם מונשם . היום הגיע בקר סגן שר פיתוח נגב גליל , איוב קרא ליכוד , הגיש בן קמע רפואה אב . קרא סיפר וואלה ! חדש על קרה , בו דבר הציל אותו קמע מוות . שנת , עת שירת קרא קצין צעיר חטיבה דרומי של צ בדרום לבנון , עליה פיקד אותו תקופה בן אליעזר , נשלח קרא פעולה ב נתקל לוחם מחבל עמד עלה . " פואד היה טוח שכמה מאייה הלכו " סיפר קרא . " הגיע חזרה דרך ראש נקרה . כשהגענו בריא ושל הוא אל א איך יכול להיות ניצל דבר כזה . אמר לו שלי נתנה אני קמע . היא אמרה אני אשמור אותו עלה כל הזמן , אף אחד לא יוכל פגוע . אז אותו תקרית קמע מונח אצל כיס " צאת בית החולים , היה קרא אופטימי סיפר כי נו של בן אליעזר אמר חווה דעת של הרופאים קובעת , כי אב ייצא מחלה ישוב לאיתנו . " היה מוכן לוותר על דבר צמוד אצל הרבה שנה כדי לראות אותו בריא שלם . הוא חסר לן חשוב לן שיחזור מהר , אחר יהיה משעמם " ר שוחח עם נו של פואד , ח קרא נתן לו ק ממשלה שוחח עם יאיר בן אליעזר , נו של חבר הכנסה מאושפז זה שבוע של דלקת ריאה . ח איוב קרא הגיע בקר , מסר לו קמע דבר הציל את חיים היה קצין תחת פיקוד של פואד
