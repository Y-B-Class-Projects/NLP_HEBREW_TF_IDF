שר כיר ליכוד מודה : " ממשלה אחדות אפשרות סביר מאוד " למרות ראש ממשלה הבהיר כי אין כוונת להק ממשלה עם מחנה ציוני , שר כיר מפלגה סבר כי " זה יהיה דבר נכון ישראל " ח אלקין : " יש סכנה ממשלה אחדות " דובר פעם הראשונה גורם מפלגת השלטון מודה תסריט הזה נשקל רצינות חשש ליכוד לקראת דוח מבקר על דיור , יפורסם היום שר כיר ליכוד מודה כי ממשלה אחדות עם מחנה ציוני היא אפשרות " סביר מאוד " ביום אחרי הבחירות . שיחה עם וואלה חדש , הסביר אמש ישי שר בכיר כי " זה יהיה דבר נכון ישראל זירה בינלאומי , שם מצפה לן אתגר מאוד לא פשוט " דבר של שר סותר את דבר ראש ממשלה , נימין נתניהו , רק לפני חודש חשבון הטוויטר שלו כי אין כוונת להק ממשלה שכ לאחר הבחירות . ממשלה ימין צר יהיה קשה התמודד עם קשיים כמו גיוס העולם נגד איראן , הדיפה לחץ אירופה התמודדות עם קו בעייתי של אובמה " הסביר שר בכיר . " בעיה האלה לא הולכה היעלם אחרי הבחירות . אנחנו צריך להיות מוכן התמודד איתן בצור טוב ביותר " זו הפעם הראשונה מערכת הבחירות נוכחי גורם תוך ליכוד מודה כך תסריט הזה נשקל רצינות . חירות כותרת נ מדיני של מרצ : הכרה מדינה פלסטיני תנאי פ דרום בוער : פוליטיקאי נוהרים נגב מבטיח עתיד ו : עד הבחירות דחה עתירה נגד ישראל היום דבר של שר ליכוד חזק את ותיו של י בית היהודי , שר נפתל בנט , אמר רק סוף שבוע האחרון כי ליכוד מתכנן לאחר הבחירות להק ממשלה יחד עם מחנה ציוני , כדי להדוף חץ בינלאומי על ישראל . עם זאת , שר בכיר הבהיר כי מהלך " יעבוד רק אם אנחנו נהיה גדול יותר הם מחנה ציוני אופן משמעותי . אם יהיה לן יתרון של מנדט , נוכל קבוע את וים מדיני של ממשלה בצור אחראי . ככל פער יהיה קטן יותר , יהיה לן קשה יותר להכתיב זאת " גב השותפים אחר ממשלה זו , שר בכיר סירב פרט מי יהיו מועמד מועדפים לקח חלק ממשלה סוג זה . גם י קואליציה , חבר הכנסה זאב אלקין , הביע דבר רוח דומה . " יש סכנה של הקמה ממשלה אחדות עם מפלגה העבודה ראשות הרצוג " הודה אלקין שיחה קיים עם תלמיד ישיבה עתניאל פורסם לראשונה אתר . חבר נס נוסף ליכוד אמר שיחה עם וואלה חדש כי אפשרות של ממשלה אחדות היא ריאלי ביותר לדעתו . " אם לא נרכיב ממשלה אחדות " אמר חבר הכנסה , " נהיה בעיה " דבר , על פי כל סקר , קואליציה תכלול את ליכוד , בית היהודי , החרדים , ליברמן חלון תמנה מנדט . " מצב כזה " אמר , " נהיה ויים כל גחמה של אל ישי , אחר תתפרק ממשלה " אותו חבר נס גם הביע חשש משי כך תיק הביטחון יימסר ליברמן : " אמנם על פי סקר יש לו רק שישה מנדט , אבל הוא יוכל לומר פשטות נתניהו תיק ביטחון , או אני הולך עם , לא תהיה לן ריר " נתניהו עצמו כתב חשבון פייסבוק שלו כי " לא תוקם ממשלה אחדות ין וב ציפי . זה לא יקר , מפנה תהום אידיאולוגי עמוק פעור ין ל " דובר הודעה דומה לזו שפורסמה חודש שעבר חשבון הטוויטר שלו , אז הוסיף : " בית היהודי הם השותפים טבעי לן יישבו ממשלה אקים , ממשלה ליכוד רחבה חזקה " את בנט דבר הללו לא שכנע , נראה כי גם ליכוד לא כולם אמין הם . ליכוד נמסר בתגובה : " דבר אינו נכון . ראש ממשלה נימין נתניהו אמר במפורש כי תהום פעור ין ליכוד ראשות לבין מפלגה העבודה כיום , כוללת נציג בולט תומך סרבנות שרת ב מסתייגים ציונות . ראשי המפלגה , ציפי , הוד כי יסכימו חלוקה ירושלים דבר אין להעלות על דעת ממשלה ליכוד . על כן , ההכרעה הפעם ברור : ממשלה ליכוד לאומי חזקה ראשות נתניהו , שבה יישב השותפים טבעי של ליכוד , או ממשלה שמאל כנוע חלש ראשות ציפי " חבר הכנסה איתן כבל העבודה , דחה את דיבור על ממשלה אחדות , אמר : " אנחנו לא מתעסקים ממשלה אחדות . אנחנו עובד להביא כמה יותר מנדט . לפני תיים התעסקות שאלה הזה עשה לן נזק . נכון רגע הזה אנחנו אמין יפנו ינו להרכיב ממשלה כי נהיה המפלגה גדול ביותר . אנחנו מנצח קרב הוא על פער " פנייה כתב אמיר תיבון : כתב עמרי נחמיאס : עדכון ראשון : שר כיר ליכוד מודה : " ממשלה אחדות אפשרות סביר מאוד " למרות ראש ממשלה הבהיר כי אין כוונת להק ממשלה עם מחנה ציוני , שר כיר מפלגה סבר כי " זה יהיה דבר נכון ישראל " ח אלקין : " יש סכנה ממשלה אחדות " דובר פעם הראשונה גורם מפלגת השלטון מודה תסריט הזה נשקל רצינות
