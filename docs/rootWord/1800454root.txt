ר ירך אפיפיור : " הפגנה יחזק יחסים " לאחר דווח כי אפיפיור ידחה ספר את האשמה כי היהודים אחראי למה ישו , כתב ר נתניהו כס קדוש על לב הפגין " דחייה טענה שווא שימש בסיס שנאה יהודי אורך שנה " בעקבות פרסום ספר החדש של אפיפיור בנדיקטוס , בו הוא מבטל את האשמה פיה היהודים הם אחראי מוות של ישו , שלח היום חמישי ר נימין נתניהו כתב רך כס קדוש . " אני ברך אותך על דחייה תקיפה ספר החדש של טענה שווא שימש בסיס שנאה יהודי אורך שנה " כתב נתניהו . ראש ממשלה אף הציע היפגש שוב עם אפיפיור : " תקווה העזה לב הפגנה יחזק את יחסים ין היהודים לבין נוצרי סביב לעולם יסייע קידום שלום פיוס דור הבאים . אני מצפה ראות שוב בקרוב להביע את העריך עמוק כ אופן אישי " אתמול דווח כי אפיפיור קבע ספר עתיד צאת אור בקרוב כי אמירות מופיעות בברית חדש , כתב של כמה שליח , מייחסות ל " את הסגרה של ישו רומאי , אין הכוונה יהודי אופן כללי , אלא רק יהודי מסוים . הספר , " ישו נצרה : שבוע קדוש , כניסה ירושלים עד התחייה " הינו רך ני ביוגרפיה כתב אפיפיור על הולדת מוות של ישו . יש פרשן מאשימים את היהודים מוות של ישו , אך אפיפיור מבטל את האשמה האלה נגד העם היהודי ספר אומר כי אין בסיס דבר . ר ירך אפיפיור : " הפגנה יחזק יחסים " לאחר דווח כי אפיפיור ידחה ספר את האשמה כי היהודים אחראי למה ישו , כתב ר נתניהו כס קדוש על לב הפגין " דחייה טענה שווא שימש בסיס שנאה יהודי אורך שנה "
