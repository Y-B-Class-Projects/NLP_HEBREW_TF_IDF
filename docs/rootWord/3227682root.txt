צולל , פסיכולוג תמונה ניצחון של חמאס : ביב בן המערכה ה ימים בחיר , קמפיין מגיעים ישורת סופי נלחם השתלט על סדר היום . נתניהו מקווה למנף את ביקור היסטורי של נשיא ברזיל להמשיך לחגוג את ההכרה אמריקני רמה גולן , אך מתיחות בדרום עשה לתדלק את קמפיין של יריב עד בנט " הולה " קצב סמבה , חגיגה קרנבל : ראש ממשלה נימין נתניהו פתח היום ראשון את תשעה ימים אחרון לפני הבחירות על השטיח אדום ב , קיבל את פנים של נשיא ברזיל , בולסונארו , הגיע ישראל ביקור ראשון תולדה המדינה . שבוע שעבר , נתניהו חזר וושינגטון , לאחר טיל מעזה גדע את חגיגה ריבונות על גולן בית לבן גזל ממנו את מלוא הכבוד תקשורתי . אחרי וושינגטון , היום מגיע עזרה חבר אחר הנשיא החדש שנוי מחלוקת ברזיל שכ הרעיף על נתניהו כבוד מלך חופשה חלום בריו דה סילבסטר האחרון , גם א הוא מתכוון הכריז על היסטוריה . אם לא שגרירות , אז פחה נציגות רשמי כלשהו ירושלים . ספוילר : גם ביקור הזה יסתיים תלונה כלשהו על התקשורת . נתניהו תכנן להינשא על ענן אופוריה גולן תוך קמפיין " יגה אחר " של ישורת האחרונה , עדיין לא הסתפק מינוף האירוע צרכי העצמה מנהיג קנה מידה בינלאומי שלא רואה את טירון ממטר . הבוקר , צפוי שיירה קמפיין ליכוד הגיע רמה קול תרועה , כמקדימה לקראת הגעה של נתניהו עצמו ישיבה סיעה חגיגי של המפלגה המשך שבוע . עוד וואלה ! : חמש רקטה נורה עבר עוטף בתגובה תקף יעד חמאס ב : " לא יהיה שר הביטחון ממשלה : " לא תהיה שר ה מוקדם לשמוח : חמאס ממתינים לראות אם שקט עז ישתלם הם אבל שבוע חצי בחיר , הישג היסטורי של נתניהו נקלע קרב גדש על סדר היום מול , לפיד גנרל . סוף שבוע גבול דרום היה גדש אירוע , אולם מבלי דבר יצאו משליטה , אך שקט מתוח עז פותח פתח מתקפה חריף על משא ומ הבנה עם חמאס , תמונה ניצחון של הניה וסינוואר יהיו חומר טוב קמפיין של כולם עד בנט ליברמן . ב עת , פרשת מניה פרשה צולל ממשיכות מקביל ספק רצף כותרת על שחיתות לסבך את נתניהו עם בן דוד מקורב רשת מסובך של אל גרסה מתדלקות את הקמפיין של חול בן . ליכוד , בתגובה , ימשיך לנסות למנף את דוח בקר המדינה על קשר המשטרה עם ממד חמישי כדי " תיק מיליון " למרות מבקר המדינה הבהיר חד משמעי כי אין חשד פלילים , וב השנייה ימשיך הנמיך את סרטון תוך רחב מכוער בוטה של נגטיב נגד כשירות של כהן ראש ממשלה להציף את עניין הזה של רוטציה ין לפיד , ציבור לא ממש יודע איך בלע אותו . עד כה זו היה אחת מערכת הבחירות מלוכלכות ו ראה כאן , אין סיבה לא חשב חלק האחרון לא יהיה משוגע יותר , טוב זה עוד עט אחור . סיפתח לקראת משוכה האחרונה , תעמוד סימן בן מול ביב , ני דים הוציא שבת הצהרה וחמניות תוקפני הם הודיע אחד שני שלא יכהנו שר ב אוצר ממשלה בא אחד של השני , כך נס הוריד סדר היום את אפשרות של ממשלה אחדות שתשמש את כל השחקנים אחר עד הבחירות . אל תתפסו אותו מילה : אחרי ייוודעו תוצאה אמת כל קלף ייפתחו מחדש , כל מה קורה עכשיו זה רק שחק מקדים פאזל פוליטי מורכב יחל אפריל , תלה מאוד מספר קטן . מה משותף כל סקר אחרון הוא ליכוד חול בן נכנס קרב צמוד מאוד על ספר מנדט יכריע את זהות המפלגה גדול , גורל יהיה תלה בעיקר מפלגה הקטנה , יכריעו מי הם יגייס רוב של ממליצים נשיא המדינה . על פנים , כל סקר יש עדיפות ברור גוש ימין אופן אוטומטי על נתניהו , כן לא ברור איפה הביטחון גמור שב מכריז הוא עומד נצח , טח דינמיקה נוכחי . אבל כחול בן משוכנעים שאם הם יהיו גדול מספיק כדי להביס אותו בנוק אאוט , לא תהיה בכלל אלה , גם מפה נאמנות פוליטי תשתנה התאם . מען המטרה הנעלה פתיחה פער של חמישה מנדט פלוס על ליכוד הם מוכן לנגוס גם שותף של גוש טבעי מפלגה המרכז שמאל , כפי הצהיר אתמול מיקי לוי פה מלא , " לא חשוב אם מפלגה העבודה תהיה או מנדט , הרבה יותר חשוב כחול בן תהיה חזק עם מנדט פער ליכוד כדי נשיא עליה להרכיב את ממשלה יהיה הפך " נתניהו , הצד השני , נזהר עד כה שלא שתת חבר מימין את מנדט , מודע היטב סכנה זה עשוי ייצר ע עדיפות גוש . אם חול בן יפתחו עלה פער של רגע האחרון , הוא עלול החליט על קמפיין " גוועלד " יעמיד אותו מצב מורכב : ככל ליכוד יגדל , כך פלגה לוויין של ליברמן , בנט , חלון דרעי נכנס כוננות ספיגה אם אחת מה לא תעבור כל מאזן גושי מאחורי נתניהו עשוי השתנות . הצליח אתמול , ראיון רינה מצליח , צמצם את נזק ה יונ יוני " למזער את נאחס של שבוע הקודם , כלל הקלטה מביכות , שמועה על חפרפרת האזנה סתר , קבלה בת נבזית עם דיווח של בן כספי מעריב על פסיכולוגי עבר לא רחוק . כל אלה נפל חומר ביד יוצר בידי אנשי ה של ליכוד , שלא בוחלים אמצעי שפל בדרך ניצחון . אפשר לא אהוב את סגנון נמוך של הקמפיין של נתניהו , אבל רמה מקצועי הוא מנהל קמפיין חד , י מתמחה ניצול הזדמנות . עומת , הקמפיין של איטי , לא מגיב מספיק מהר אירוע , לא על נפילה של נתניהו . שלוש שעה אחרי הראיון ה של יוני לוי , ליכוד כבר היה סרטון אוויר גחיך אותו , תיים אחרי פרסום דוח מבקר על ממד חמישי כירי ליכוד התייצב תקף אותו מסיבה עיתונאי . מה לוקח ליכוד שעה ספור בלבד לוקח צד נגדי ארבעה ימים רק ביום רביעי נזכר כחול בן סרטון מלגלג על ריאיון מגומגם של נתניהו קרן מרציאנו . לשם השוואה , הקמפיין יעיל חזק של מפלגה העבודה , ידע לנצל היטב את התבטאות אומלל של נתניהו התבכיין שלא קיבל סיקור עיתונאי תוך שעה , גם על ההזדמנות הכניס מירי רגב לחגוג על חשבון חוצות אשקלון . לעומת העבודה , כחול בן לא מפספסים הזדמנות פספס הזדמנות . מצודה זאב סינן שבוע הנאה " רק ימשיכו כך " אבל אם דינמיקה ין ביב בן תמשיך להמריא , ארבעה ימים אחרון עשה להתרחש זרימה של מנדט שול עבר ליכוד חול בן כל מפה עשה השתנות .
