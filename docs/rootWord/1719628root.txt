ר וועדת טירקל : ברק היה האחראי ב אמר פתח עדות פני עד טירקל כי ערב נסיעה ל טיל את אחרית טיפול משט עז על כתף שר הביטחון . נתניהו סרב ענות על אל וועדה האם נשקל חלופה השתלטות על משט צילום : גדעון צנטנר . עריכה : גדי ינסטוק ראש ממשלה נימין נתניהו העיד הבוקר ני פני עד טירקל בדיקה אירוע משט עז אמר כי טרם נסיעה ל , טיל את אחרית אירוע משט על שר הביטחון אהוד ברק . דבר , " נחתי מי אחראי על טיפול משט על כל היבט יהיה שר הביטחון , ירכז את הנושא יפעיל את שרי שביעייה את נושא הסברה . הוא היה אחראי ניקוז כלל נושא " כמו כן , סרב ראש ממשלה השיב פומבי מספר אל של חברי וועדה , וב השאלה האם נ חלופה השתלטות על מרמרה . שאלה ה מיגל דויטש האם ישיבה שביעייה קדמה משט עלה הערכה על עוצמת ההתנגדות צפוי משט אמר נתניהו כי דבר עלה שיחה אגבית בדיון , אך סרב פרט על דבר פומבי אמר כי יפרט דלת סגור . פתח דבר אמר נתניהו כי מדינה ישראל ו פעל התאם חוק הבינלאומי ו צ על סיפון גיל לב בלתי רגיל מילוי שימה הגנה על עצמו מפנה סכנה חיים ממשי " ראש ממשלה אמר חבר וועדה כי הוא מאמין זו תרד שורש אמת , " אני ראשון מופיע פני עדה מכובד זאת , רב אחר יופיעו אחרי אני מאמין תבררו את כל פרט " הוסיף נתניהו . עדות של ראש ממשלה נימין נתניהו מול וועדה ראשות שופט בדימוס יעקב טירקל , היא אמור הראשונה שלב גבייה העדויות . נתניהו צפוי העביר מהלך היום את עדות מלא תוך גם את חלק יישמעו דלת סגור . מהלך שבוע הקרוב יעידו מול וועדה גם שר הביטחון אהוד ברק ו גבי אשכנזי . ביום רביעי התכנס חברי וועדה יחד עם משקיף הזרים הגיע ישראל לקראת שמיעה יותיהם של ראש ממשלה , שר הביטחון ו , דן ין שאר גבייה העדויות . וועדה נמסר כי כבר לפני שבוע העביר רכז וועדה , הושע גוטליב , שלושה אישים בכיר את נושא מרכזי יהם יתבקשו התייחס . גוטליב סיכם את ועד תן העדויות עם לשכה אישים יותיהם יישמעו בית הארחה ע רבי ירושלים . אתמול התייחס נתניהו סמכות עד טירקל אל מול עד מ ה בדיקה אירוע משט . נתניהו עדכן ישיבה ממשלה שבועי את שר על מינוי של ד יוסף נציג ישראלי וועדת מ ה אמר כי " " תחילה חודש יולי אישר ממשלה את הרחיב סמכות וועדה , זאת בעקבות דרישה של ראש וועדה , השופט בדימוס יעקב טירקל . התאם הרחבה סמכות , יכול וועדה זמן עד ול העדויות תחת בועה . עם זאת , הרחיב סמכות לא חל על
