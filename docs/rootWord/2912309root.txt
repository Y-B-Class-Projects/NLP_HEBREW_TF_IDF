ה נכנס רוני אלשיך טקס השבעה : " כאן כדי שרת " אלשיך קיבל את דרגה טקס משרד ראש ממשלה , לאחר תיים התקיים טקס נוסף מטה ארצי של המשטרה . ה נכנס נאם : " פתאום הרגיש הו כובד שליחות מוטלת על כתף " ניצב סאו : " מאחל לו הצלחה , יום כונן חיי הארגון " נתניהו : " מצפה רפורמה " ה מיועד רוני אלשיך מושבע הבוקר חמישי תפקיד . סגן ראש ה לשעבר היה אמור היכנס תפקיד לפני שבועי , אולם בר את רגל יום לפני כן כניסה רשמי תפקיד נדחה . האירוע מצוין שני טקס . ראשון נפתח שעה משרד ראש ממשלה ירושלים , שם מוענקות לו דרגה על יד ראש ממשלה נימין נתניהו וה ביטחון פנים גלעד ארדן . השני י תיים לאחר כן מטה ארצי , אז ייכנס אופן רשמי תפקיד . טקס הענקה הדרגה אמר אלשיך הלצה : " הושבע טקס צנוע , על פי פרוטוקול . יים הם לא חלק טקס . רגל שמאל נשבר על מנה הבטיח את ניסה רגל ימין " קריאה נ אלשיך : ניצב חשוד עבירות ין יאריך את חופשה מ הודיע על פרישה המשטרה : " הגיע לאן ש צבאי תפקיד כיר חדש : המהפכה מתכנן אלשיך משטרה אלשיך הודה תמיכה קיבל כל אורך דרך . " אז גמלה לבי ההחלטה לקח על עצמי את משימה שמש כ , נחשף מנה גדוש , הן רחש לב תוך משטרה ישראל , קרב בכיר כמו קרב דרג טח , הן רחש לב של פורש המשטרה השקיע ב את מיטב ותיהם והן רחש לב של ציבור כול , תוך חבר קרוב , אך בעיקר אינספור אזרח מהשורה . אחרון , סתם כך פוגשים ב סופרמרקט , מזהים את פני לאחר אישור ראש ממשלה הוסר פיקסלים , כמובן מרגישים צורך אנושי , אחל הצלחה לחזק את יד ? י " דבר , " פתאום הרגיש אופן מוחשי , עד כמה חזק הקשר ין חברה מתוקן ריאה וב שטר איכותי משרתת אותו , עד כמה באמת תמה נחוץ ההצלחה של המשטרה נגזר כך , הו כובד שליחות מוטלת על כתף " עוד הוסיף אלשיך כי " פרק הזמן עמד רשות להכיר עומק את הארגון מבפנים , למדתי עד כמה המשטרה מעוניינת ב ? שינוי , פחה כמו מי בחר ב להנהיג אותו . מצא קרב מפקדי המשטרה בכלל קרב סגל הפיקוד כללי פרט שותף אמת גיבוש תכנית פעולה ממוקד נחוש , אותו נניע יחד , כדי הצליח משימה לממש את ייעוד המשטרה , כי אזרח ישראל אין שטר אחר " נוסף על כך , ציין אלשיך כי " אנחנו כאן כדי שרת . תפקיד שרת את האזרח נורמטיבי שומר החוק , לאפשר לו ביטחון אישי מיטב צד מלחמה נחוש בלתי מתפשרת במי בחר להגדיר עצמו עבריין " טקס השני מטה ארצי של המשטרה הוסיף אלשיך : " לפני זמן קצר מאוד , הצטרף רשמי משפחה שוטר , אף כי אוד חש שוטר רגע ראשון שב גמלה לבי ההחלטה נטלתי על עצמי את משימה שמש כ . אז הוטל על משטרה ישראל אחרית ביטחון פנים , המשטרה מתמודדת עם אתגר ניהולי מקצועי , שלא מוטל על ארגוני שטר דומים בעולם . המשטרה הוכיח שבוע אחרון כי הקשר אתגר זה , היא נמצא כל מקום עבר האזרח בטחונו אישי , מובן הכי מילולי של מילה . האתגר לן יהיה עבר האזרח כל תחומי חיים . לא רק ביטחון פנים , לא רק זירה סמוי עין של מלחמה נחוש פשיעה , אלא גם פתרון עיות יומיום של האזרח , כולל אלה נוגעות איכות חיים . כול כאן כדי שרת ואין אני ספק שיש עם מי " הוסיף . על פרישה ניצב צמרת הפיקוד אמר : " נידרש מלא מחדש מהירות את ותינו חסר . שמחה אני גלה אחרון עוד עוד אזרח רוצים התגייס שורה המשטרה , תוך אידאל . חימם את לבי שבוע אחרון להיחשף גם עוצמה התנדבות של אזרחי ישראל שורה המשטרה , תחום רב גוונים של שירות אזרח " אלשיך התייחס גם פגיעה אמון ציבור ארגון : " אין ספק משטרה , וודאי מדינה דמוקרטי , לא יכול לתפקד ללא אמון ציבור אותו היא משרתת . רוח פגש שטח קרב שוטר מפקד , מן גם רוח התנדבות של אזרח כן שינוי מגמה חש לשכה גיוס , סמן נקודה אור חשוב נתיב העולה של אמון ציבור " נתניהו אמר טקס כי כניסה של אלשיך תכלול ינויים גדול . " יש צורך רפורמה , אתה תרפא את מחלה . טכנולוגיה גם סייבר , זה נושא חשוב , יש צורך רפורמה לוקחות חשבון את כל מכלול . קנית לך שם של פקד נחוש יצירה שלא חושש יזום . אני מצפה שני ינויים גדול מאוד עבודה המשטרה חייב כלל את כל אזרחי ישראל , יש כאן דין אחת אוכפת את הביטחון ומדינה אחר אינו אוכפת את הביטחון . זה לא יכול הימשך " אמר . הוא הוסיף : " משטרה ישראל היא אבן יסוד . אין שטר בלי שטר . אני בקר שטח חודש אחרון אני נפעם התגייסות של שוטר שוטר . הם פועל נגד פעילות עוין מלחמה טרור . משימה מוטלות עלה אין אח רע מדינה אחר . אני חש מחויבות עמוק הגן על ביטחון של אזרחי ישראל על כך אני אומר בשם העם כול תודה רב " שר ביטחון פנים , גלעד ארדן , פנה מהלך טקס אלשיך אמר : " אתה היום שומר חומה ירושלים . אנו זקוק משטרה מקצועי איתן " דבר , קיום חברה חופשי תלה משטרה בעל עמוד שדרה ערכי , שואבת עקרונות צדק של חבר מדינה . זו שלא תטה אוזן לאלו מבקש רק את טובה פרטי על פני זו של חברה , וכ תשאב את כוח נורמה ערכי נעלות יאפיינו את אנשיה מפקד , בוא אכף את קיום החוק על ציבור " עוד הוסיף ארדן כי " שרת מפקח כללי של משטרה ישראל היא חשוב מדינה , ואין פלא תהליך בחיר של ה החדש לווה תהודה ציבורי גדול . נימה אישי אומר תודה לאל תהליך הסתיים בדרך טוב שבה הסתיים " דבר , " ההחלטות התקבל כובד ראש לאחר התלבטות קשה , אל מול גודל
