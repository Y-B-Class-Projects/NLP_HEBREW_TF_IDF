רשימה חיסול של ממשלה י חלול ישראל מנסה מנוע דיון ציבורי על סדר עדיפות מעוות לה , ממשיך לשתוק בעוד ליברמן ממשיך מסע חיסול של כל מי מפריע לו . פרח חושב שאנחנו בדרך מלחמה א אמריקני לשעבר , הנרי , קח מעט שבעים שנה עד התנצל על הפקרה יהודי גרמני נשלח מחנה ריכוז כמובן התנצלות מאוחר לא סייע מיליון שילם חיים . אומה העולם , חש בושה מ על הובלה היהודים תא הגזים החליט חלק את פלשתין להק שתי דינות , שנה אחר כך , , החליט שאמנה בינלאומי זכות אדם היא חובה אוניברסלי . אז עד היום , מוכיח העולם שוב שוב הוא מסוגל להפקיר בן אדם , פעם אפריקה , פעם בוסניה , פעם , פעם סין כך גם ישראל . מדי פעם , העולם חוזר עסוק דרפור , סין או בוסניה כך , מתפנות גם לעתים מדינה מ שם לב מתרחש ישראל . אך המדינה הוקם בעקבות הפרה זכות אדם חמור ביותר היסטוריה אנושי מבקשת שלא יעסקו ב . המדינה הבטיח עבר שלא תשתוק על רדיפה יהודי , מפתחת מנגנון תייק השתקה על יבוש גניבה מולדת של עם אחר . דין הפכה מיליון פליט , מחזיקה מיליון תחת טון יבוש אלף בתי מעצר , מבקשת עתה חקר גם את " סייען טרור " היהודים . ים הם מתעוררת מחאה של אזרח מהשורה נגד שר העביר מיליארד חרדי מתנחל , מקום כבאית , יוצאים חבר ממשלה מתקפה חסרת תקדים על ארגוני זכות אדם . מדינה שבה מתעוררים רק בן אדם נשרף חיים יער כרמל , רוב דומם כאשר תוקף את מתריעים שער . מדינה שבה ראש ממשלה הבטיח כבר שלא תהיה יותר " זקן פרוזדור של בית החולים " מת תינוק זקן מסדרון בתי החולים , מבלי יהיו מתריעים שער . ארגוני זכות אדם הוקם כדי לקח אחרית לא חכה אסון , להתריע שער כאשר ממשלה פוגע זכות אדם . הם קבל סף מדינה קרן הסיק מסקנה ממלחמת העולם השנייה , מנסות מנוע את מלחמה בא . מי נלחם ארגון זכות אדם הן דינות מפרות את אותו זכות : כך קורה היום סין , רוסי , איראן , תוניס , סודאן גם ישראל . מנהיגות שלא מסוגל לדאוג מערכת כיבוי אש , מיטה אשפוז בתי החולים , דיור הוגן מחיר סביר צעיר , אוכל ילד עני , היא מנהיגות חלול . מנהיגות הזאת מנסה מנוע שוב שוב דיון ציבורי על סדר עדיפות מעוות לה : היא לא רוצה נדבר על תשעה מיליארד שקל זורמים אלף מתנחל , או על מיליון שקל משרד שיכון עביר חבר מאבטח את אות מתנחל מזרח ירושלים , גם לא על מיליארד עובר מוסד ש תומך אל ישי , סוחטים את כספי מיסים ציבור העובדים משלם . הם לא רוצים נדבר על שמירה על היערות , לא על מי ים מזוהמים , על כנרת מתייבשת , על הכיבוש , על אפליה נשים אזרח ערב . אחרי הרג עז אנשים , הם גם ילד , אחרי פגע קשר ידידות של ישראל טורקיה חיסל את יחסי של ישראל , עת הם נס ייצר מלחמה אזרח עם " האויב מבפנים " עתה אחרי הוכיח הוא " לא בין ערבי " כנראה גם לא רוסי , אנגלי , אמהרית או עברי שר עניין לום מתפנה לחסל גם את ארגוני זכות אדם . אחרי יסיים לחסל את ערב את " סייען טרור " הוא יטפל במיכאל איתן , בן בגין וב מרידור . ליברמן מוביל מלחמה אזורי , מלחמה אזרח עם ערב ישראל , כל מי יפריע לו יחטוף . נתניהו , ישי אהוד ברק הם שותף שתיקה מעשה . אחרי ש " עזתים , ערבי , עובד הזרים , פליט , מתגיירים , רפורמי , ארגון זכות אדם יגיע תורם של ארגון סביבתי , האחיות שימח על מצוקה אשפוז , מרץ אוניברסיטה נשים יתקוממו על תקיפה מיני אפליה , גם תור של דני רוזן . כך נוהג פשיזם . הוא מחסל את שכן , את המיעוט , את מהגר אחר כך מתפנה טפל שי ניצן , משנה פרקליט המדינה , כפי טיפל יצחק רבי , ראש ממשלה לשעבר . פרח הוא מנהל רכז מוסאוא זכות אזרח ערב ישראל . רשימה חיסול של ממשלה י חלול
