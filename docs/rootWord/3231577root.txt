מה נשתנה לילה הזה ? נתניהו חזק יותר על חשבון שאר ה ממשלה נתניהו החמישית , החל כבר סימן שכרון כוח הופיע . לריבלין הוא הכניס , כתב אישום הוא לא מפחד , קול נגד מינוי אפשרי של לוי משפט רגב ביטחון פנים ויים דווקא חזק אותו . אופוזיציה ? הבטיח תהיה מרור , אבל בינתיים היא הרבה יותר ח הם חשש רגע הזה יותר מכל , שני אולי דמיין אותו לגמרי אחר , אבל כל ה התנקז סוף של דבר לטקס ממלכתי סטנדרטי אפילו די משעמם . רגע שב נשיא המדינה ראובן ריבלין העניק בנימין נתניהו את כתב המנדט הרכבה ממשלה , שררה בעיקר תחושה חזק של דה וו ככה זה , כנראה , כש קורה פעם החמישית . ניגוד תופעה פסיכולוגי שבה אנחנו רגיש היינו מצב מסוים כבר פעם זה לא באמת נכון , מקרה הזה זו אמת אמיתה . אחרי מערכת חירות סוער חזרנו בדיוק אותו מקום : אותו ראש ממשלה , אותו קואליציה , אותו שותף שינוי פרסונלי קל של בנט שקד הוחלף רף בצלאל דיוק אותו עיות חוק גיוס , עבודה שבת , פסק התגברות החלה ריבונות עומדים ראש סדר היום של משא ומ קואליציוני , אל תופתעו אם בקרוב נשמע שוב על חוק עונש מוות מחבל . אז מה כל זאת נשתנה לילה הזה מכל לילה ? נתניהו חזק יותר מאי פעם , ליכוד עם הישג שיא היסטורי שלא נרשם אז , כל שאר השחקנים מסביב יותר חרדי דתי או יותר מגומדים מוחלש . סימן שכרון כוח הופיע כבר ביום ני בוקר , כ התייעצות של סיעה בית הנשיא נשלח שלושה ליכודניקים הכי גידף את ריבלין : שר תיירות יריב לוי , שרת התרבות ספורט מירי רגב , ו ודוד ביט , בחינה תריס משתחצן . בדיוק כמו ההחלטה מנה את מקורב נתן אשל , הורחק שירות ציבורי , חבר רשמי צוות משא ומ קואליציוני . או האיתותים על הכוונה מנה את איש אמון לוי או י איחוד לאומי משרד משפט ולשים את אנשי חצר רגב אמיר אוחנה משרד ביטחון פנים תקשורת . לא ברור אם נתניהו רק משחרר בלון ניסוי כדי בחון את עוצמת תגובה באמת נחוש לכת עד סוף , אבל תדרוך עלום שם של גורם משפטי משטרה על סכנה מערכת אכיפה החוק רק מתדלקים את בייס תאוריות קונספירציה על ו " מקום יהיה הם אפקט מצנן הם דווקא ויים חזק את נתניהו צדיק דרך . פסק התגברות היא רק ההתחלה יש כאן הצהרה כוונה ברור גב מלחמה הוא מתכוון נהל נגד פרקליטות , המשטרה , בית המשפט עליון תקשורת קדנציה קרוב אמצעות נאמן אנשי אמון . הפיל גדול חדר בית הנשיא שעל אף אחד לא דיבר היה השעון משפטי מתקתק : עד סוף השנה אמור להיות פה שימוע אולי גם כתב אישום , אבל נתניהו לא מתכוון לתת זה עצור אותו . נישא על גבי תמיכה ציבורי חסרת תקדים , " רצון בוחר " ישמש מגן חסינות שלו כל דרך בית המשפט . סיטואציה קואליציוני חדש , צפרדע חסינות תיבלע גרון של כל השותפים גם בלי נתניהו יעשה ום דבר , אף אחד הם פשוט לא יפיל אותו . שר האוצר שה חלון , היה יחיד התבטא נגד המשך כהונה של נתניהו תחת כתב אישום , חזר בחירות עם מנדט בלבד , ללא משקולת של שון אזניים יכולת לאיים הפלה ממשלה או הטיל וטו יציל את מערכת המשפט מאימת ה . אם מאמץ הכניס את כול סיעה משותף עם ליכוד יבשילו , הוא יוותר אפילו על יכולת ערער את יציב ממשלה להשאיר אותו מתכונת צר של , יכשיר נתניהו את דרך . נתניהו כבר חגג עם עדן בן זקן את ניצחון גדול אבל לפני מברכים על מוגמר , ויים עוד כמה בועה מפרך של משא ומ , כאב ראש גדול שלו יהיה ענייני דת ומדינה . לוין אשל יש עד מאי למצוא פתרון יצירה יאפשר זאב כבש גור זה עם זה , לגשר על דרישה של החרדים שנה את חוק גיוס מול איום של ליברמן לכת אופוזיציה אם ישנו בו פסיק נקודה . חרדי יש מנדט לעומת בלבד של ליברמן , אבל הוא ההבדל ין עלה תקום תיפול ממשלה . גם חלון לוקח את הזמן : למרות מגע מתקדם לכאורה של מקורב על חזרה הביתה ליכוד , הוא התחמק כל שבוע להגיע פגישה עם נתניהו העדיף לצפות במשחק כדורגל ברצלונה , אפילו לא הודיע מי ישב טעם צוות משא ומ קואליציוני . ליכוד מוטרדים גרירה רגל של חלון מודאג לא פחה מליברמן לא צפוי יכול לעשות קטע רגע האחרון . חלון ליברמן מכחישים שיש ב תיאום , אבל יכול ייצר ציר של שיתוף פעולה בלוק ימין יכול אזן את התחזקות סיעה חרדי להקשות על חלק דרישה של משא ומ . לא טוח זה אפילו רע נתניהו , יהיה גוש מימין יעזור לו עמוד חרדי סוגיה מסוים מרוצה קדנציה . גם אם ליברמן חלון ימשיכו שחק אותו קשה השגה , גם אם דרישה של יתגלו כקשות פיצוח , ההערכה רווח היא סוף של דבר , גם אם דרך תהיה ארוך פתלתל , כולם ירד נתניהו ישביע סוף מאי את ממשלה החמישית . אבל כיוון מדובר אותו שחקן ו קואליציה , הנה ספוילר מתבסס על דה וו : זה יימשך עד דקה לא ייגמר דקה אחת קודם . זמן נתניהו קיבל מריבלין את המנדט להרכיב את ממשלה החמישית , עשה להכשיר את דרך כהן עם כתב אישום , אופוזיציה חדש השתרר דממה חרישי . חול בן , מפלגה השלטון אלטרנטיבי , לא פצה פיה , מפלגה העבודה לא נמסר תגובה , רק י מרצ תמר זנדברג היה צדיק יחידי סדום הזכיר נתניהו לא רק רכיב ממשלה אלא גם מחפש חסינות תציל אותו מצב משפטי . יאיר לפיד הבטיח למרר את חיי נתניהו אופוזיציה אבל אם שפט שבוע ראשון זה של ממש . על ארבעה נים דיבר הגדה שבוע אחרי הבחירות רביעייה צמרת של חול בן פחה או יותר התפוגג אוויר . י המפלגה , בן , הגיח ראיון קצר עובדה , גבי אשכנזי נשיא כדי הוריד פרק את רעיון ממשלה האחדה , לפיד בילה פריז בוערת ותיו של שה יעלון נעלם ליל . אפשר הבין מה אחרי ימים של קמפיין תיש , סוער רצוף עלייה מורד הם היה זקוק התאוששות הלם קרב . אבל זמן קוקפיט עשה איש איש ביית , הקמפיינרים אסטרטג יצאו תקשורת סיבוב ניצחון , או שמא מסע חיסול , רפש התחיל צאת החוצה . מצד אחד , יש במה התפאר מנדט זה הישג שיא מפלגת אופוזיציה שלא נראה בשנים האחרונות , אבל התוצאה סופי לא היה מספיקה מהפך ול צריך סמן את האחראי . ניצחון אב רב אבל כישלון הוא יתום אף אחד לא רוצה הוא יהיה רשום על שם . זמן כולם דיבר על אגו אמון של , לפיד גנרל , מסתבר סכין חד נשלף על יד אנשי הקמפיין . מונח לקוח עולם ניתוח לאחר מוות , מה יועץ של עש שבוע חשף את מה קרה מאחורי קלע היה לא פחה מדמם . אם לתמצת בקצרה את עיקרי עלילה , מתנהלת בעיקר ציר , אנשיו של ראשון מאשימים את השני החרבה הקמפיין אולי אפילו כוונה תחילה . רונן צור , אסטרטגי של , התנגד חיבור עם לפיד מלכתחילה , הוזז גלל ניהול הקמפיין , עכשיו חזיר נקמה : שורה של ראיונות , תדרוך התבטאות אז הבחירות טען לפיד אנשיו סוקר מארק למן פוליטי שלום שלם הוביל קמפיין מקרטע חלש אחראי בלבדית הפסד " פושע " הוא פספוס היסטורי . צור רץ לכת נתניהו על ראש טרגט את בוחר ימין , אך צוות של לפיד התנגד דגל אסטרטגיית המפלגה גדול מהות אבה את מנדט גוש . גם את אחרית נאום ניצחון שלא היה צור זרק פתח של לפיד שיחה סגור אף העלה את אפשרות הוא שלח את להתבזות על במה כדי לעשות כוון . אחר של , ישראל בכר , התראיין חדשה , ותלה את מנדט חסר הקלטה , שודר שבוע אחר שבוע אותו חשף את כפל לשון חרדה של י חול בן . בכר , זה מלכתחילה לכת על איחוד עם לפיד הצביע גם על דרמה הגדולות מאחורי קלע מהמורה משמעותי . לשון : " לאחר איחוד , התקבל החלטה שלא מפטרים אנשים החלטה אנושי . אני חושב מקצועי היה צריך החליט אחר " לא טוח כמה זה כובד מקצועי צד של ספינולוגים לכ את כביסה מלוכלך לא טוח סיבוב ה של מועילים מיוחד , אלא רק חזק את דימוי שלו בובה מהונדסת על חוט . רוב יועצים , לבד בכר , לא אמור המשיך א אופוזיציה , אבל חשדנות בסיסי לפי לפיד ככל הנראה תחלחל עמוק תוך קדנציה . כבר שבוע דלף טוויטר של עמית סגל תכתובת ווטסאפ פנימי קבוצה ח שבה ה רשימה של קרא תיגר על לפיד הביע התנגדות קו הוא מוביל מול החרדים . אם הדלפה הקלטה ימשיכו א קמפיין כנסת , עיתונאי כמובן נברך על כך , אבל רמה ניהול שוטף זו תהיה תקלה קשה . חול בן עומדת פני אתגר אופוזיציוני עצום : גם בלי יועצים , קוקפיט יש מספיק אגו , שלושה ר צריך התרגל חיי ח אפור וועדות מליאה . נוסף , עם פחה ח חדש , ניסיון פרלמנטרי של אנשי יש עתיד הוא מצרך הכרחי כדי ייצר אופוזיציה לוחמת . עדיין לא ברור איך תעבוד ביחד סיעה מאוחד מי בדיוק יחזיק מושכה , אבל לפני לפיד מתחילים למרר ממשלה את החיים הם צריך סדר את מה קורה אצלם בית . נתניהו ליכוד יחפשו כל עת , טבע , הכניס טריז תוך אלטרנטיבה לבצע הפרד שול , אתגר הכי גדול של לפיד יהיה שמר על חול בן מאוחד . עם פרסום תוצאה סופי של הבחירות , ראשי ימין החדש נפתל בנט איילת שקד הצטרף רשימה מכובד ארוך של נפגע קורבן שלא רדה את חירות . כישלון ימין החדש רבדים רב , החל נטישה בייס ציונות דתי , קמפיין ביזארי בוטה וכלה צעקה גוועלד של נתניהו שת את כל קול פלגה ימין . בנט שקד שותקים מנסים שמר על פרידה אבל גם על דינמיקה ב עוד יסופר רב . למרות ההנהגה משותף , התבוסה לא מתחלקת שווה שווה אלא מוטלת בעיקר פתח של בנט , ניווט את ההגה ניהל את הקמפיין עד רגע האחרון נלחם על קול וקול , בעוד שקד מסומנת מי תצא אירוע סוף של דבר מנצחת . התרסקות המיזם משותף משחררת את שקד כבלי חסות של בנט מאפשרת לה צאת דרך עצמאי חדש מבלי להיות מואשמת תקיעה סכין גב שותף לה עוד הרבה לפני נולד ימין החדש , דוד ביט פתח ע את דלת ליכוד רווחה , לא מן נמנע מיד אחרי החגים מישהו יבחין ב גישה טופס התפקדות מצודה זאב . בנט לא קיבל הזמנה זו אף אחד , אבל מוקדם להספיד גם אותו . למרות על פנים תאונה הדרכים משותף הוא יצא נזק גדול אילו היא עם מכה קל פגוש . הבחירות דרמטי סוער יספקו פרשן חוקר אקדמיה עוד הרבה חומר דיון מעמיקים , אבל ניתוח הכי פשטני שאלה : מה הביא נתניהו את ניצחון , מתבונן על זהות מועמד התמודד ולו , מידה כריזמה נוטפת הם . פוליטיקה מודרני הולכת מתבססת על מנהיג אישיות , מיוחד מקרה של ישראל הערצה נתניהו , פרסונה של ראשי מפלגה היא בעל חשיבות מכריע . מקרה של בנט שקד מעורר תרגיל מחשב עניין משעשע של ספקולציה אחור " מה היה קורה איל " התחשב פופולריות של שקד ממד הערצה אליה חוג ימין , אפשר דמיין מה היה קורה איל היא היה עומדת ראשות ימין החדש מקום בנט ? האם היא היה צליחה לסחוף קהל אחר או גרום הם הישאר בית , או שגם היא היה נסחף קשית מנדט של נתניהו ? גם צד מנגד אפשר אתגר את דמיון להציב קושיה משעשע מבט של בדיעבד : עמוק עם ראש מים תחילה קמפיין הבחירות , קיבע את עצמו תור מועמד של מחנה " רק לא ביב " ראשות ממשלה , אילו גבי אשכנזי , היה האחרון הצטרף גדול גוש המרכז , הסתפק מקום רביעי , תחת מי עבר היה פקוד . אשכנזי חרש את שטח בחיר התגלה , מצופה , מגנט פופולרי וקים פריפריה מצליח הקסים גם מעוז ליכוד , אבל סוף לא היה גרוש לירה תביא מהפך כפי קיווה , להסתפק ספסל אופוזיציה . נעצום את עין נדמיין מה היה קורה איל מול נתניהו , מקום וב , היה תמודדים אשכנזי שקד וה זה היה קשה עלה הבקיע את ניצחון היסטורי . כדי להשלים שלישייה קושיה , נצלול אחורה זמן ליולי , פריימריס על ראשות מפלגה העבודה . מה היה קורה אם י המפלגה אבי גבאי לא היה נצח עמיר או הרצוג היה ראשות מפלגה העבודה ? האם הם היה מצליח זהות את פוטנציאל של או אשכנזי בעוד ועד לדאוג הוא ינותב מפלגה של עצמו , וב מונעים עבודה התרסק מעמד מפלגה טון מפלגת נישה ? נבהיר : דובר ספקולציה חסר ממשות משמעות , נקודה מחשבה או תסריט מדומיינים , שעשוע בלבד כבוד חופשה פסח חיי הנגאובר אחרי הבחירות . חג שמח ! מה נשתנה לילה הזה ? נתניהו חזק יותר על חשבון שאר ה ממשלה נתניהו החמישית , החל כבר סימן שכרון כוח הופיע . לריבלין הוא הכניס , כתב אישום הוא לא מפחד , קול נגד מינוי אפשרי של לוי משפט רגב ביטחון פנים ויים דווקא חזק אותו . אופוזיציה ? הבטיח תהיה מרור , אבל בינתיים היא הרבה יותר חרוסת
