ה גבה את המלצה המשטרה : " תיק נתניהו נוהל לפי הספר " מנדלבליט יצא הגנה הארגון אמר כנס אוניברסיטה ת כי פרסום המלצה להעמיד לדין את נתניהו עבירות שוחד , נעשה תיאום מלא א : " לא תהיה אני כל עי מקצועי הגיש כתב אישום נגד נתניהו " כמו כן , סירב התייחס אפשרות ש חשוד פרשה משפטי ממשלה אביחי מנדלבליט גיבה אמש חמישי את התנהלות המשטרה חקירה ראש ממשלה נימין נתניהו , אמר כי " שבוע הועבר פרקליטות תיק החקירה פרשות . חקירה נוהל לפי הספר אופן שב צופה רשות החוק טפל תיק סוג זה , חתירה מתמיד גילוי אמת " דבר משפטי ממשלה נאמר רקע תגובה של ראש ממשלה , פרידה ין המשטרה פרקליטות הדגיש את תפקיד של משפטי ממשלה מי יקבע סוף של דבר מה יעלה גורל תיק . המשטרה המליץ להעמיד לדין את נתניהו עבירות של שוחד , מרמה הפרה אמון תיק תיק . מנדלבליט הדגיש נאום דוקוא את הקשר ין המשטרה פרקליטות , הבהיר כי החקירה נוהל תוך שיתוף פעולה מלא ין אגף חקירה מודיעין משטרה , לבין פרקליטות המדינה משפטי ממשלה . " אני שומע על מתיחות ניסיון פלג ין הגופים . אנחנו עובד שיתוף פעולה מלא הפך כל אבן להוציא את אמת אור . אין איש על החוק ואין אדם חסין מפנה . זה לא אומר אין ב חילוק דעה . צריך הבין זה סדר דבר , כך מתנהלות חקירה " דבר נאמר כנס אגודה ידיד של אוניברסיטה תל אביב . עוד וואלה ! מחכה לו : האם אביחי מנדלבליט חומר נכון ? ארדן קרא מנדלבליט : בדוק את הטענה מעקב אחרי חוקר נ סיציליאני : אחרי אלשיך תקשורת , נתניהו סימן את לפיד אויב קיומי ה הוסיף כי " כול חלק אותו מערכת אכיפה החוק . זיהה נושא דורשים בירור העמקה , עשה כל שביכולתנו כדי להעמיק הם לברר אותו . צד זאת , עלה חקירה נושא צדדי אינו צדיק בירור שדה פלילי , נמנע להשקיע הם זמן משאבים . זיהה גורם עדות שלו חיוני על מנה הגיע חקר אמת , לא חסך מאמץ על מנה גב את עדות " מנדלבליט נושא לראשונה דבר לאחר פרסום המלצה המשטרה להעמיד לדין את ראש ממשלה נימין נתניהו בגין שוחד . עוד הבהיר מנדלבליט , כי פרסום ההקלטות תואם א אופן מלא . " פרסום הודעה המשטרה ביום ישי נעשה התאם הנחיה ה רלוונטי . דבר נעשה אישור אישור פרקליט המדינה ראש , נדרש מקובל . בהחלט קיים אפשרות כי אגב בדיקה חומר החקירה , יידרשו השלמה חקירה המשך . זו דרך של כל חקירה . דבר נובע מ ברר את החשדה להוציא את אמת אור " " אני דוחה כל תוקף טענה כאילו מי גורם מעורב פעל תוך שיקול זר אינו מקצועי . כל הגורמים פעל פועל נאמן טון החוק מדינה ישראל " אמר מנדלבליט . " אני גם ער עניין ציבורי הרב , חשוב אני להדגיש שכל ההחלטות התקבל יתקבלו מבוססות רק על ראיה דין " ה הוסיף : " עתה תחל מלאכה לימוד משפטי של החומרים , כדי הכריע אם מתגבש על סיס סיכוי סביר להרשיע . ראיה יישקלו , סוגיה משפטי מתעוררות ילובנו אורה . תהליך יתקיים קודם כל פרקליטות מחוז תל אביב , לאחר כן יועברו פרקליט המדינה שי ניצן . לאחר כן יובא מכלול חומר החקירה העמדה שונות ל . זה תהליך מקצועי , אין בו קיצור דרך אסר יהיו . כוונת קיים את ההליך ללא עיכוב מיותר . רק ראיה ידברו רק הדין יקבע . זו חובה אזרח המדינה " מנדלבליט נשאל ריאיון עם עיתונאית נחמה דואק מיד לאחר נאום , אם לא תהיה לו עי הגיש כתב אישום נגד נתניהו , אם יידרש כך . " אני מכיר את ראש ממשלה רק רקע מקצועי . עולם לא היה ביית קיסריה . בית בלפור היה כמה פעם בנסיבות מקצועיות . וודאי יהיה קשה אני הגיש נגד כל ראש ממשלה , כל שר , זה לא דבר נע . אבל אני מחפש את אמת אם אלו יהיו ראיה זה יהיה הדין , לא תהיה אני עי " אמר . תשובה שאלה אם לא ראוי נתניהו יכריז על נבצר , השיב ה : " זה עניין שלו שלב הזה , לא תפקיד הציע לו דבר אלה . הוא חף פשע כל עוד לא הוכח אחר . ברור שיש שקל המלצה אבל אין הן תוקף חוקי . תוקף חוקי יש רק מקבל החלטה , בוודאי שאנחנו לא שלב הזה עכשיו . נגיע נשקול את דבר " כמו כן , סירב מנדלבליט התייחס אפשרות ראש ממשלה חשוד פרשה צולל . ראש ממשלה נימין נתניהו אמר אתמול לפני יציאה גרמני כי הוא רעייה " נפעם , אין מילה אחר , תמיכה , אהדה חום אזרח כה רב ישראל מגלים כ " רקע פרסום המלצה נגד . נתניהו הוסיף : " אני אמשיך שרת אותו , את המדינה לן אחריות , נחישות תוך תחושה ביטחון גדול , כי אני טוח אמת סופה נצח " שלשום שלח שר ביטחון פנים גלעד ארדן ל מנדלבליט , ובו קרא לו למצות את בדיקה הטענה פיהן בוצע עקב אחרי חוקר בכיר משטרה היה עורב חקירה ראש ממשלה נימין נתניהו . ארדן התייחס לטענות עלה ריאיון העניק ה רוני אלשיך ל " שם טען כי " גורם רבי כוח " עקב אחר חוקר שטר . שר האוצר שה חלון אמר אמש כנס חיפה כי " יש אדם אחד חוק הסמיך אותו קבוע עניין ראש ממשלה נימין נתניהו והוא משפטי ממשלה אביחי מנדלבליט . אני קורא צד ימין שמאל אפשר רשות החוק לעשות את העבודה כדי קבל את התוצאה טוב ביותר " עדכון רשון : ה גבה את המלצה המשטרה : " תיק נתניהו נוהל לפי הספר " מנדלבליט יצא הגנה הארגון אמר כנס אוניברסיטה ת כי פרסום המלצה להעמיד לדין את נתניהו עבירות שוחד , נעשה תיאום מלא א : " לא תהיה אני כל עי מקצועי הגיש כתב אישום נגד נתניהו " כמו כן , סירב התייחס אפשרות ש חשוד פרשה צולל
