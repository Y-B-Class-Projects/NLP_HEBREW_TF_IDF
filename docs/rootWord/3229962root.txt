ין : אופציה של חלון ליברמן קלקל נתניהו את תמונה ה דרמה סביב ימין החדש אחוז חסימה עיצום , ראשי כול ישראל יתנו מגיעים ל קואליציוני וטים לאחר קמפיין געוואלד של נתניהו . חבירה משותף עשה החזיר הם את שליטה על העצמאות איבד עם פרסום תוצאה ה סילון , עם צוות דייל מפואר , י ישראל יתנו אביגדור ליברמן המריא אתמול רביעי חופשה קצר בירה אירופי החביבה עלה , ינה . זה נהג ידע מוכר של ליברמן נשם אוויר פסגה בינלאומי אחרי חירות , על אחת כמה כמה אחרי הקמפיין הקשה אותו שרד בקושי עם חמישה מנדט , קל חומר אחרי מבצע שאיבה מנדט של רגע האחרון של נתניהו ליכוד יוון , ין היתר , ישיר בייס רחוב רוסי . אחרי ניסה לחסל אותו , ליברמן לא ממהר לשום מקום בטח שלא תוך קואליציה של מי עמד ראש . הוא השאיר א רשימה דרישה מפואר עלה על מטוס . זה הזמן הנות שניצל וינאי משובח סיגר טוב . דרמה של הבחירות עדיין נמשך , סימן השאלה בדיקה סביב ספירה קול חייל משאירים את משחק לגמרי פתוח עם כל מיני תרחיש אפשרי שונה אופן מהותי משמעותי . אם בנט שקד יעברו סוף של דבר את אחוז חסימה , נתניהו תהיה אפשרות להרכיב קואליציית חלום של ח עם הרבה שחקן קטן הוא יכול לתמרן ב . פרט סיעה חרדי אף פלגה לא תהווה שון אזניים שעל תקום או תיפול ממשלה גם אחרי הגשה כתב אישום . אם ימין החדש לא יעברו , הוא כנראה להסתפק קואליציית בלבד ב כל אחת סיעה יש את כוח מנוע ממנו את הרכבה ממשלה או פרוש להפיל אותו עת צורך . ליברמן הוא נעלם גדול של משא ומ קואליציוני עומד פני ליכוד . למרות הוא גדל חצר של נתניהו מקור הוא שר מבשר של מפלגה השלטון , מרוצה שנה העצמאות שלו הפך אחד קלף הכי פחה בלתי ניתן חיזוי מערכת פוליטי . הוא כבר הוכיח עבר אופוזיציה לא מפחיד אותו : אחרי חירות הוא החליט הישאר יום לפני הצגה ממשלה , הוא הותיר את נתניהו מתפתל עם קואליציית במשך קצת יותר שנה . זאת עוד , נובמבר האחרון ליברמן לא היסס עזב את משרד הביטחון להתפטר מחאה על מדיניות עז , מה עצם הוביל סוף של דבר הקדמה הבחירות . עת , הוא מציב דרישה לא פשוט עבר נתניהו קואליציה הוא מתכנן להק תתקשה להכיל אותו הוא עזב את ממשלה מחאה על העביר מיליון דולר חמאס עכשיו הוא דורש קודם כל שינוי חד מדיניות לפי עז תנאי ב קואליציוני . בעוד נתניהו מערכת הביטחון דוחפים הסדרה ארוכה טווח , ליברמן רוצה הכרעה החלטה על יציאה מבצע צבאי נרחב רצועה . מכשול השני , מורכב לא פחה , הוא החזית נוקשה הוא מציב מול מפלגה חרדי , כולל דרישה ווטו עניין דת ומדינה התעקשות על אישור חוק גיוס , והוא לא מתכוון אפשר דרעי יצמן ממש את כוח מוגבר קלות . ליברמן דורש את תיק הביטחון , אבל לא הכרח באמת ממהר חזרה קומה קריה עשוי העדיף דווקא משרד אחר שב ישב כבר עבר משרד . יוקרה פוליטי של משרד הביטחון התברר מפוקפק , כך גם יכולת שלו השפיע על דיניות הביטחון שאותה נתניהו מותיר ריכוזיות יתר תיאום הדוק עם צ , מטוס סילון עם דיפלומט זר הוא יכול למצוא עצמו יותר יכולת ביטוי אשר עם גנרל . תמהיל דרישה יש הוסיף את עז וותיקות של ליברמן כל מ קואליציוני : חוק עונש מוות מחבל , פנסיה עולה אחריות על העלייה הקליט , אבל גם אלה גם חלוקה תיק הן זוטות , מחלוקת על חמאס , עיקר החזית מול החרדים , הן מהמורה הגדולות עומדות לפני דיל סגור עם ליכוד . אפשר סמך על ליברמן הוא ינצל כל רגע במשחק לפני ימהר התמסר יד של נתניהו הוא לא ישכח קלות את קמפיין געוואלד רוסי על ראש שלו . זה מה הוא עושה מחיה . בעוד שליברמן סיים רק מנדט פחה בחירות קודמות , שה חלון ניצל עור שיניו מקרבות געוואלד של נתניהו , רק בקושי התייצב על אחוז חסימה , עוד לא ברור אם יסיים ארבעה או חמישה מנדט . אחרי קדנציה ארוך יחסי מוצלח אוצר הוא לא הצליח זכות אמון ציבור כמו כל מפלגה הקטנה נפל קורבן קרב סוער ין נתניהו חטף טראומה קשה . מצב הזה , הוא אפילו לא שון אזניים , אין לו הרבה קלפי מיקוח לקראת משא ומ קואליציוני , והוא חשוף מיוחד ניסיון שכנוע של נתניהו בכיר אחר ליכוד חזר הביתה מפלגת האם . ההישג גדול בחיר , מנדט למרות חקירה למרות גנרל , הביא תחושה אופוריה ליכוד הוליד מחדש את אחד חלום גדול של נתניהו חלום המפלגה רפובליקן . צמד מיל אמריקני הזה מבטא מהות מיזוג של פלגה לוויין ימין תחת כנף ליכוד מטרה יצור מפלגה טון אחד גדול עלה גם אחרי הבחירות קודמות , כמו אז , חלון שוב חוזר ניסיון לפת אותו חבר סיעה משותף כנסת . כבר בליל הבחירות גדעון סער שלף שידור חי את יוזמה החזרה של חלון הביתה מיזוג כול תוך ליכוד , ומ בכיר אחר ליכוד מהדהדים את מסר עקביות ניסיון קדם את עניין עוד לפני המלצה אצל הנשיא . דלת נפתח כחלון היא נכסף נחשק תאפשר לו התמודד על נהג המפלגה ביום אחרי נתניהו , חיבוק מפתה מיוחד אור ניסיון צורב שלו הלך בדרך עצמאי , אך המשמעות מיידי שלו היא חיזוק כוח של נתניהו עיקר נסיגה הצהרה פיה ראש ממשלה לא יוכל כהן אחרי יוגש נגד כתב אישום . חלון יקבל מפתח ניסה חזרה ליכוד אבל יחזור להיות עוד אחד שרי ליכוד ממתינים ליום אחרי נתניהו בינתיים בולעים כל ביקורת או עמדה עצמאי על התנהלות , ניגוד הצהרה יעניק במו יד נתניהו חסינות מפנה הפלה ממשלה גם אחרי שימוע . מבחינה פוליטי , עם ארבעה מנדט אין כחלון יכולת מיקוח טוב מיוחד , עם חמישה היא משתפרת . אולם מבחינה ערכי מהותי הוא גורם מאזן יחיד נתניהו יוכל סמך עלה תוך קואליציה חרדי , ימני , קיצוני , ול הוא לגמרי עדיין במשחק . ין , בן גביר , ש מחוזקת יצמן שמר על כוח , חלון מייצג ימין ממלכתי מתון , והוא עצם הפך עלה התאנה הימני ליברלי קואליציה ימין של נתניהו . קדנציה קודם , מפלגה כול למה יוזמה ליכוד נועד פגוע מערכת המשפט , מבקר המדינה , משטרה תקשורת , שילם על כך מחיר . הרכבה החדש , בן גביר מסתמנים מהלך אימים עוד יותר אשר אצל שקד וב . חלון יכול חזרה ליכוד אך גם שחק עם קלף : הנחה נתניהו צריך אותו כדי אזן את נוכחות דתי חרדי מוגבר קואליציה חדש , יש לו כל זאת עוד מרווח תמרון . אם בנט שקד גוש ימין אכן יעמוד רק על מנדט כל זאת אין ממשלה , אפילו רוב של ממליצים , והוא חוזר עמדה כוח של שון אזניים . אם לא הוא יכול להיצמד ליברמן יעשה לו את העבודה שחור . הכול עדיין פתוח שלל רב של תרחיש , אבל כל קרה נתניהו יצא ידו על עליון בחיר יחסי הכוחות עם שותף טבעי . אחרי קמפיין אגרסיבי שאב את מנדט מחל רגע האחרון , הוא חוזר קואליציה אותו הרכב כמו קודם הוא חרדי מחוזקים כל שאר השחקנים מוחלש , מקרה של בנט שקד אולי אפילו גרוע כך . עת הוא מקווה מצפה כולם ירוצו אל מגומדים שמחה ששון , אבל אם כחלון יהיה את אורך נשימה של ליברמן מהלך הכי חכם בחינה רגע עשוי להיות חבירה ישראל יתנו בלוק טכני של ח . זה יאפשר שניים נהל את משא ומ קואליציוני עמדה כוח משמעותי מהותי , להוות שקל נגד התחזקות סיעה חרדי . ביחד הם יוכלו דרש שלושה , אולי אפילו ארבעה תיק , להבטיח עצמו מינוי בכיר אוצר , או ביטחון . חילוני של ליברמן מתיישב עם צורך של חלון שמירה על תקציב המדינה מפנה היכנעות כסף בן ישיבה , כוח משותף יאפשר כל אחד הם הכניס תנא הסכם קואליציוני יבטיחו שיש הם יד על ההגה חבר לא מחריבים את צהל מערכת המשפט . שני תרחיש או בלי חלון ליברמן ביחד נתניהו לא תהיה ממשלה , בלוק משותף של יאפשר שניים שנה את דינמיקה של הקמפיין להשיב יהם את עוצמה עצמאות , מקרה הזה טוב שני מן אחד . ין : אופציה של חלון ליברמן קלקל נתניהו את תמונה ה דרמה סביב ימין החדש אחוז חסימה עיצום , ראשי כול ישראל יתנו מגיעים ל קואליציוני וטים לאחר קמפיין געוואלד של נתניהו . חבירה משותף עשה החזיר הם את שליטה על העצמאות איבד עם פרסום תוצאה
