ר נתניהו : " ידיעה אחרון עיוות כותרת מכוון זה לא עיתון זה פראבדה " תקף חריף של נתניהו על ידיעה אחרון בגלל עמוד ראשון של אתמול , טען כי ר השווה ין הורי חייל עומד עכשיו משפט אשמה הריגה לבין הורי הדר גולדין . ר אמר כי זה " חיבור קרי , זה כבר לא עיתון , זה פראבדה " תקף חריף של נתניהו על ידיעה אחרון בגלל עמוד ראשון של העיתון אתמול , כותרת היה " השוואה מקוממת של נתניהו : התקשר ורי חייל יורה כמו התקשר הורה יהם נפל או נעדר " מרבית של עמוד ראשון הובא מאמר של נחום ברנע , נפתח מלים " השוואה ין הורה של חייל עומד עכשיו משפט אשמה הריגה לבין הורה של הדר גולדין אורון אול זועקת שמיים " " לא ביקש לעשות ום השוואה " אמר הבוקר ר נתניהו ראיון לאסף ליברמן ב , " לא תיתכן ום השוואה ין מצוקה של הורה מצב אלה או אחר מצוקה נוראי , סבל נורא של הורה שכול . אני אומר לך את זה ידיעה אישי , לצערי . לא היה כוונה ערוך השוואה זאת . אמירה לא הכי מוצלח , זה נכון . אז קודם כל אני הבעה צער אם מישהו פירש את דבר הללו בדרך מעוות , אבל אותו נשימה אני אומר לך שיש מישהו עיוות את זה מכוון . עיתון מסוים , אם הוא קורא עצמו עיתון , ידיעה אחרון , לוקח את כל עמוד השער . מה הוא עושה עם עמוד השער ? כתבה מעוות , פשוט כתב קר רשע מנסה יצור את הקבלה הזה כוח . אז אני אומר : אני לא התכוון לעשות את זה אני מצטער אם מישהו פירש את זה כך , אבל הם כן התכוון וודאות יצור את חיבור שקרי הזה , זה מקובל , זה כבר לא עיתון , זה פראבדה " זהו סיבוב נוסף קרב מתמשך ין ר נתניהו ל אחרון " בעל שליטה נוני מוזס . תחילה יולי עלה פוסט עמוד פייסבוק של ליכוד , שגם נתניהו שיתף עמוד פייסבוק שלו , שב הם האשים את העיתון פרסום קרי ניפוח של הוצאה ביקור ר אפריקה . הכתב מדיני של העיתון , איתמר אייכנר , הגיב פוסט משל כתב " אני עיתון עומדים מאחורי מספר פורסם " ר נתניהו : " ידיעה אחרון עיוות כותרת מכוון זה לא עיתון זה פראבדה " תקף חריף של נתניהו על ידיעה אחרון בגלל עמוד ראשון של אתמול , טען כי ר השווה ין הורי חייל עומד עכשיו משפט אשמה הריגה לבין הורי הדר גולדין . ר אמר כי זה " חיבור קרי , זה כבר לא עיתון , זה פראבדה "
