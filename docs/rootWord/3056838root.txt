טיגריסים , גיבור על תינוק : מנהיגי העולם סוחפים את ה ממשלה הוד הוא פוליטיקאי עם הכי הרבה עוקב , טראמפ אחור אבל כולם יכול למד מאובמה . ללא עניבה , נס פוליטיקאי העביר סר קצר קליט סוחפים א מיליון מרחב העולם . איפה ממוקם נתניהו ? ראש ממשלה הוד , נרנדרה מודי , נשיא ארץ הברית , דונלד טראמפ , הם המנהיגים פופולרי ביותר אינסטגרם . אולם שניים עוד דרך ארוך עד יגיעו כמות עוקב נכסף של הנשיא לשעבר ברק אובמה . מודי ספר עוקב גבוה ביותר בעולם מיליון ואחריו טראמפ עם מיליון עוקב . אובמה , היה מנהיג ראשון בעולם פתח חשבון רשת החברתית מבוססת תמונה , יש עת חשבון פרטי יותר עוקב מלשניים יחד מיליון . נשיא ארץ הברית לשעבר יש עת חשבון פרטי יותר עוקב מלשניהם גם יחד מיליון עוקב . אובמה גם היה זה סיפק לעולם את תמונה גרף הכי הרבה ייקים שנה האחרונה צילום ליל כל קדוש בית לבן אוקטובר אחרון , זיכה אותו מיליון חצי ייקים תגובה . טראמפ ניצח תחרות ווידאו נצפה ביותר , עם סרטון זכה יותר מיליון צפייה של בת איוונקה משחקת עם בן תיאודור בית לבן . ראש ממשלה נימין נתניהו , מקרה שתהיתם , הרחק מאחור . מחקר שב יותר חשבון אינסטגרם של ראשי דינות שרי דירג את חשבון פרטי של נתניהו מקום בלבד , עם אלף עוקב . חשבון רשמי של נתניהו , תחת שם , נמצא מקום עם קצת יותר אלף עוקב . אינסטגרם היא רשת חברתי הושקה מאוחר יחסי , שנת , ארבע שנה אחרי טוויטר שש שנה אחרי פייסבוק . אז , הצטרף רשת קהל משתמש מוערך מיליון איש . אינסטגרם היא פלטפורמה שם דגש על תמונה , טקסט ב אינו האלמנט המרכזי . היא הפך אמצעי תקשורת אטרקטיבי מוקד עבר פוליטיקאי , מחפשים העביר סר קצר קליט , לעתים משוחררים כבלי עניבה . חברה יחסי ציבור אמריקני ורסון מרסטלר , מפעילה פרויקט רחב של ניטור פעילות פוליטי דיפלומטי טוויטר רשת חברתי נוסף , מיפה את העולם פוליטי אינסטגרם גילה כי ראשי דינות יש פרופיל אישי . כמו כן , ביותר דינות יש חשבון רשמי אישים גוף ממשלתי . שנה חולף , יותר מיליון איש רחבי העולם נחשף יותר אלף תמונה סרטון הועלה על יד מנהיג או משרדי ממשלה . רוב המנהיגים לא נהלים את חשבון עצמו , משאירים את העבודה צלם רשמי נבדל סגנון רמה חשיפה שהם מאפשרים תכנים מאחורי קלע . שימוש אינסטגרם מאפיין בעיקר את אירופה המזרח תיכון , פופולרי פרט מדינה . מודי , כבש את המקום ראשון דירוג אינסטגרם , הוא כנראה איש חזק ביותר בעולם כיום בחינה שליטה רשת חברתי . הוא גם מנהיג שלו ספר עוקב גדול ביותר טוויטר פייסבוק יותר מיליון איש . מקום השני דירוג , אמור , נמצא טראמפ , מעלה ממוצע יותר שני חצי פוסט ביום , שנה חולף פרסם פוסט . אפיפיור הוא מדורג ישי לאחר שיק את חשבון שנה שעברה והגיע מיליון עוקב . חשבון רשמי של בית לבן , שלו מיליון עוקב , נמצא מקום רביעי . ישראל , נוכחות פוליטי אינסטגרם דל למדי . נתניהו היה מנהיג הראשונים בעולם פתח חשבון אינסטגרם אפריל , ומ הפריד את הפעילות שני חשבון נפרד אחד רשמי מוקדש תמונה מעונב של נאום פגישה רשמי , שני של ליכוד , ובו מגיחים עת עת צילום סרטון של נתניהו רעייתיו , ילד כלב קאיה . דירוג ישראל , נמצא נתניהו מקום ראשון עם אלף עוקב , ואחריו ח יאיר לפיד יש עתיד , עם אלף . ל בית היהודי נפתל בנט עשרה אלף עוקב , אך חשבון לא הועלה תמונה אז ספטמבר שעבר . ל אופוזיציה יצחק הרצוג אין חשבון פעיל כלל . שנה חולף נתניהו כפיל את ספר עוקב שלו שני חשבון , אך לעומת אינסטגרם , נוכחות שלו זירה בינלאומי יותר בולט רשת חברתי אחר . מחקר דומים של אותו חברה יחסי ציבור מיקם את נתניהו מקום דירוג המנהיגים פופולרי פייסבוק , סרטון העלה נובמבר האחרון , שב ירך את הנשיא טראמפ אחרי ניצחון , היה הסרטון רביעי הכי נצפה שנה שעברה עם יותר אלף צפייה . גם זירה פוליטי הישראלית פייסבוק נתניהו שולט ללא עוררין , והוא ניצב מקום ראשון עם קרוב שני מיליון עוקב , לפיד וב מקום השני , פער גדול , עם אלף עוקב . מה עוד נוגע ישראל ? חשבון אינסטגרם פופולרי ביותר הוא חשבון רשמי של המדינה , מופעל בידי משרד תחת שם . הוא נמצא מקום עם יותר אלף עוקב . חשבון מפעיל מחלקת דיפלומטיה דיגיטלי משרד ערבית , , הגיע מקום עם עוקב . חשבון של משרד נמצא מקום עם עוקב , חשבון של הנשיא ראובן ריבלין ניצב מקום עם עוקב בלבד . טיגריסים , גיבור על תינוק : מנהיגי העולם סוחפים את ה ממשלה הוד הוא פוליטיקאי עם הכי הרבה עוקב , טראמפ אחור אבל כולם יכול למד מאובמה . ללא עניבה , נס פוליטיקאי העביר סר קצר קליט סוחפים א מיליון מרחב העולם . איפה ממוקם נתניהו ?
