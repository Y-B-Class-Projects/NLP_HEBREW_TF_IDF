טובה . כך או כך , ין אם אפיק יישמר וב אם ינדוד אפיק אחר , ייאלצו שלם הרבה סף . מחיר עבר שמירה על אפיק מוערך מיליון שקל . גם קרה של נדידה אפיק אחר , את השקיע סף פרסום לספוג הפסד תוצאה איבוד צוף . מקרה זה , לא יכול להאשים את ראש ממשלה או גלגל את אחרית עבר הרשות השנייה , ממנה זכה חודש אחרון שורה של הקלה בדרך קבלה רישיון . זכר , על מנה הקל על לעודד את זכיין נוסף , קשת רשת , לעבור שידור רישיון , הוחלט על הקלה משמעותי מחויבות תוכן . סכום נדרש השקעה תוכן ירד מיליון שקל מיליון שקל שנה . גם נושא ערבות קיבל הקלה משמעותי , כאשר נדרש שלם מיליון שקל מקום מיליון שקל . לפני חודש , לאחר תן הקלה , ישיבה מועצה הרשות השנייה טלוויזיה רדיו , נקבע כי שיר בקשה אפיק בקשה רישיון שידור משך שנה . זאת מידה יצליח הסיר שלושה חסם עד מאי : הסדרה מבנה הבעלות , עמידה התחייבות כספי ערבות סכום של יותר מיליון שקל . כמו כן , על להסדיר את חוב עבר דמי ביד . עם קבלה הצעה הגיש קבוצה , נראה יוכלו הציג רשות השנייה את הסדרה מבנה הבעלות מנייה מוחזקות נאמנות . כמו כן , יוכל עמוד התחייבות הכספיות . גם סוגיה חוב עבר דמי ביד נמצא , כך נראה , בדרך פתרון יקל על . חוב , המלצה נתניהו עומד על מיליון שקל , יחזור בדיקה מחודש על יד שר האוצר שה חלון . את עמידה תנא אמור הגיש רשות השנייה , תפתח בדיקה צורך תן רישיון , לוח זמן מאוד צפוף , כן ביוני פוקע זיכיון של . אם לא יתעוררו עיות מיוחד , יוכל קבל רישיון יולי . אולם אז יתחיל קרב על אפיק . אפיק שייך מדינה . על פי החוק כאשר זכיין עוברת שדר רישיון עליה עזב את אפיק . איל שלוש זכיין טלוויזיה היה עוברה שדר רישיון , היה עומדים רשות אפיק עד . מקרה של , הוא רשאי הישאר אפיק כל עוד זכיין לא עוברה רישיון נשאר אפיק . אלא אישור המשיך לשדר אפיק נמצא אמור יד של שר התקשורת נתניהו . גם אם המלצה של נתניהו תהיה היענות בקשה , עדיין ייאלצו התמודד עם עי נוסף : תג מחיר יקבע עבר שימוש אפיק . מקרה יתקבל אישור הישאר אפיק , תוקם עדה מיוחד תבדוק את השלכה כלכלי תקבע מה סכום על שלם , סכום נאמד כמה מיליון שקל . בינתיים , עד נושא אפיק לא יוסדר , הישארות של אותו אפיק לאחר קבלה רישיון ייחשב הפרה . מוזר עד עכשיו , למרות כל דיון הצהרה על עבר רישיון לא טיפל נושא אפיק לא הגיש כל בקשה . הרי לא נשאר זמן , יולי הוא עובר רישיון . יש אפשרות בעל חדש של יירצו לוותר לגמרי על אפיק , תדמיתי הציג חדש לגמרי , אבל גם עבר אפיק אחר דורש קמפיין פרסום , הסברה , השקעה כדי שמר על צוף . גם מעבר אחר הטמעה דורשים פרק זמן של
